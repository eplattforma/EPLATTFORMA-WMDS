Step 1.1 — Add a helper sync_attribute_dimension(session, attr_no)

Add this function near your other DW sync helpers (below _upsert_dimension is a good place). Adjust import names to your project structure (you already import DwAttribute1..6 and call_ps365 here).

from sqlalchemy.orm import Session
from requests import HTTPError

ATTRIBUTE_DIM_MODELS = {
    1: ("DwAttribute1", "attribute_1_code_365", "attribute_1_name", "attribute_1_secondary_code"),
    2: ("DwAttribute2", "attribute_2_code_365", "attribute_2_name", "attribute_2_secondary_code"),
    3: ("DwAttribute3", "attribute_3_code_365", "attribute_3_name", "attribute_3_secondary_code"),
    4: ("DwAttribute4", "attribute_4_code_365", "attribute_4_name", "attribute_4_secondary_code"),
    5: ("DwAttribute5", "attribute_5_code_365", "attribute_5_name", "attribute_5_secondary_code"),
    6: ("DwAttribute6", "attribute_6_code_365", "attribute_6_name", "attribute_6_secondary_code"),
}

def sync_attribute_dimension(session: Session, attr_no: int, page_size: int = 100) -> int:
    """
    Sync one PS365 attribute dimension (attr_no 1..6) into dw_attribute{n}.
    Returns number of rows upserted.
    """
    if attr_no not in ATTRIBUTE_DIM_MODELS:
        raise ValueError(f"attr_no must be 1..6, got {attr_no}")

    # Resolve model classes from globals by name (keeps code compact)
    model_name, code_field, name_field, secondary_field = ATTRIBUTE_DIM_MODELS[attr_no]
    model_cls = globals().get(model_name)
    if model_cls is None:
        raise RuntimeError(f"Model {model_name} not found in datawarehouse_sync.py globals()")

    upserted = 0
    page = 1

    while True:
        # Your codebase uses call_ps365 already; keep consistent with the same auth pattern.
        # Endpoint name is "list_attributes".
        resp = call_ps365(
            "list_attributes",
            {
                "attributeNo": attr_no,
                "page_number": page,
                "page_size": page_size,
            },
            method="GET"
        )

        rows = resp.get("list_attributes") or resp.get("data") or []
        if not rows:
            break

        for r in rows:
            # Map fields defensively (PS365 naming sometimes differs)
            code = r.get("attribute_code_365") or r.get("code") or r.get("attributeCode")
            if not code:
                continue

            payload = {
                code_field: str(code).strip(),
                name_field: (r.get("attribute_name") or r.get("name") or "").strip(),
                secondary_field: (r.get("attribute_secondary_code") or r.get("secondary_code") or r.get("other_name")),
            }
            payload["attr_hash"] = _compute_hash(payload)
            _upsert_dimension(session, model_cls, code_field, payload)
            upserted += 1

        session.commit()
        page += 1

    return upserted

Create a quick CLI script (or admin route) to normalize zone names to codes:

def normalize_invoice_item_zones():
    rows = InvoiceItem.query.filter(InvoiceItem.zone.isnot(None)).all()
    for r in rows:
        z = (r.zone or "").strip()
        if not z:
            continue
        # Convert known names to code
        code = attr3_name_to_code.get(z.upper())
        if code:
            r.zone = code
    db.session.commit()
