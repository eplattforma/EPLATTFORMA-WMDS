function selectAllFiltered() {
  if (!lastTotal) {
    alert("No customers in current filter.");
    return;
  }
  selectionMode = "all_filtered";
  document.querySelectorAll('.rowCheck').forEach(x => x.checked = false);
  document.getElementById('checkAll').checked = false;
  updateSelectedInfo();
}

function clearSelection() {
  selectionMode = "rows";
  document.querySelectorAll('.rowCheck').forEach(x => x.checked = false);
  document.getElementById('checkAll').checked = false;
  updateSelectedInfo();
}

C) Quick Assign chips (click group name → assign immediately)
1) Render chips after metadata loads

In loadMeta() (after you load groups), add:

Add these functions:

function renderGroupChips(groups) {
  const el = document.getElementById('groupChips');
  el.innerHTML = '';

  // Clear chip
  el.appendChild(makeChip('(Clear group)', ''));

  // Existing groups
  groups.forEach(g => el.appendChild(makeChip(g, g)));

  // Use typed group
  el.appendChild(makeChip('(Use typed value)', null));
}

function makeChip(label, value) {
  const b = document.createElement('button');
  b.type = "button";
  b.textContent = label;
  b.style.padding = "7px 10px";
  b.style.border = "1px solid #ddd";
  b.style.background = "#fff";
  b.style.borderRadius = "14px";
  b.style.cursor = "pointer";

  b.onclick = async () => {
    const typed = document.getElementById('assignGroup').value.trim();
    const rg = (value === null) ? typed : value;
    await quickAssign(rg);
  };

  return b;
}

2) One function that assigns based on selection mode

Add:

async function quickAssign(reporting_group) {
  if (selectionMode === "all_filtered") {
    const msg = `Assign ${lastTotal} filtered customers to group: "${reporting_group || '(clear)'}" ?`;
    if (!confirm(msg)) return;
    await assignAllFiltered(reporting_group);
    return;
  }

  const codes = getSelectedCodes();
  if (!codes.length) {
    alert("Select customers (checkboxes) OR click 'Select ALL Filtered'.");
    return;
  }

  await assignSelected(reporting_group, codes);
}

D) Modify assign functions to accept reporting_group parameter

Replace/implement these in your script:

async function assignSelected(reporting_group, codesOverride=null) {
  const codes = codesOverride || getSelectedCodes();
  if (!codes.length) return alert('Select at least one customer.');

  const r = await fetch('/admin/customer-reporting-groups/api/assign', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ reporting_group, customer_codes: codes })
  });

  const j = await r.json();
  if (!r.ok) return alert(j.error || 'Error');

  selectionMode = "rows";
  await loadPage(currentPage);
  alert(`OK. matched=${j.matched}, updated=${j.updated}`);
}

async function assignAllFiltered(reporting_group) {
  const r = await fetch('/admin/customer-reporting-groups/api/assign', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      reporting_group,
      apply_to_filter: true,
      filters: filtersPayload()
    })
  });

  const j = await r.json();
  if (!r.ok) return alert(j.error || 'Error');

  selectionMode = "rows";
  await loadPage(1);
  alert(`OK. matched=${j.matched}, updated=${j.updated}`);
}
