6) Do not rely on CREATE TABLE IF NOT EXISTS alone if the table was partially created

Update update_forecast_ordering_schema.py to include this helper:

import logging
from sqlalchemy import text
from app import app, db

def _column_exists(conn, table_name, column_name):
    result = conn.execute(text(f"""
        SELECT 1
        FROM information_schema.columns
        WHERE table_name = :table_name
          AND column_name = :column_name
        LIMIT 1
    """), {"table_name": table_name, "column_name": column_name})
    return result.fetchone() is not None

def _add_column_if_missing(conn, table_name, column_name, ddl):
    if not _column_exists(conn, table_name, column_name):
        conn.execute(text(ddl))
        logging.warning(f"Added {table_name}.{column_name}")
