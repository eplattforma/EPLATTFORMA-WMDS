3) Frontend: “Share to mPrint” button (Web Share API)

Add a button that fetches the file and opens the Share sheet.

Implementation (use PNG endpoint)
<button id="btnPrintThermal">Print (BIXOLON)</button>

<script>
async function shareThermalFile(url, filename, mime) {
  const res = await fetch(url, { cache: "no-store" });
  const blob = await res.blob();
  const file = new File([blob], filename, { type: mime });

  // iOS/Android: this opens Share sheet where "BIXOLON mPrint" appears
  if (navigator.canShare && navigator.canShare({ files: [file] })) {
    await navigator.share({ files: [file], title: filename });
    return;
  }

  // fallback
  window.open(url, "_blank");
}

document.getElementById("btnPrintThermal").addEventListener("click", async () => {
  // IMPORTANT: never call window.print()
  const url = "/receipts/{{ receipt.reference_number }}/print.png";
  await shareThermalFile(url, "receipt.png", "image/png");
});
</script>
