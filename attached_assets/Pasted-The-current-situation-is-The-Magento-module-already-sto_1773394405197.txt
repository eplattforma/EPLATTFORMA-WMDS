import os
import requests
from datetime import datetime
from typing import Any, Optional


MAGENTO_BASE_URL = os.getenv("MAGENTO_BASE_URL", "").rstrip("/")
MAGENTO_BEARER_TOKEN = os.getenv("MAGENTO_BEARER_TOKEN", "")
MAGENTO_ACTIVITYLOG_ENDPOINT = os.getenv("MAGENTO_ACTIVITYLOG_ENDPOINT", "").lstrip("/")


def fetch_activity_logs() -> Any:
    if not MAGENTO_BASE_URL:
        raise ValueError("MAGENTO_BASE_URL is not configured")
    if not MAGENTO_ACTIVITYLOG_ENDPOINT:
        raise ValueError("MAGENTO_ACTIVITYLOG_ENDPOINT is not configured")

    url = f"{MAGENTO_BASE_URL}/{MAGENTO_ACTIVITYLOG_ENDPOINT}"
    headers = {}

    if MAGENTO_BEARER_TOKEN:
        headers["Authorization"] = f"Bearer {MAGENTO_BEARER_TOKEN}"

    resp = requests.get(url, headers=headers, timeout=30)
    resp.raise_for_status()
    return resp.json()


def _parse_rows(payload: Any) -> list[dict]:
    if isinstance(payload, list):
        return payload

    if isinstance(payload, dict):
        for key in ("items", "data", "logs"):
            value = payload.get(key)
            if isinstance(value, list):
                return value

    raise ValueError("Unexpected response format from activitylog API")


def _to_dt(value: Any) -> datetime:
    if not value:
        return datetime.min

    text = str(value).strip()

    try:
        return datetime.fromisoformat(text.replace("Z", "+00:00"))
    except Exception:
        pass

    for fmt in ("%Y-%m-%d %H:%M:%S", "%b %d, %Y %I:%M:%S %p"):
        try:
            return datetime.strptime(text, fmt)
        except Exception:
            continue

    return datetime.min


def get_customer_last_login(
    customer_id: Optional[int] = None,
    email: Optional[str] = None,
    ps365_code: Optional[str] = None,
) -> Optional[dict]:
    payload = fetch_activity_logs()
    rows = _parse_rows(payload)

    def match(row: dict) -> bool:
        if customer_id is not None and str(row.get("customer_id")) != str(customer_id):
            return False
        if email is not None and (row.get("email") or "").strip().lower() != email.strip().lower():
            return False
        if ps365_code is not None and str(row.get("ps365_code") or "").strip() != str(ps365_code).strip():
            return False
        return True

    filtered = [row for row in rows if match(row)]
    filtered.sort(key=lambda row: _to_dt(row.get("last_login_at")), reverse=True)

    if not filtered:
        return None

    row = filtered[0]
    return {
        "customer_id": row.get("customer_id"),
        "email": row.get("email"),
        "first_name": row.get("first_name"),
        "last_name": row.get("last_name"),
        "ps365_code": row.get("ps365_code"),
        "last_login_at": row.get("last_login_at"),
        "last_logout_at": row.get("last_logout_at"),
    }
