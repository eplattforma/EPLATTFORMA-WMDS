You’re very close, but as pasted this will break and your PDF will not print cleanly on SPP-R310 from iOS/Android browsers. Here’s a precise review + the concrete changes I would tell Replit to make.

1) Critical problems in your pasted code (must fix)
A) Syntax / file corruption

At the end of send_cod_receipt() you have:

return jsonify({'error': str(e)}), 500   from io import BytesIO


That from io import BytesIO sitting after the return is a syntax error (looks like two files got concatenated).
✅ Fix: Move all the PDF code (BytesIO, ReportLab functions) into a separate module, e.g. services/thermal_receipt_pdf.py, and import it.

B) Your PDF width is wrong (165mm)

In build_delivery_receipt_pdf() you set:

PAGE_W = 165 * mm


That is way wider than an 80mm thermal roll, and iOS/Android will “fit to page” and scale unpredictably → tiny text, weird line breaks, inconsistent margins.

✅ Fix: Use 80mm page width with ~4mm margins (content ~72mm).

C) 48 columns + font size 15 will never fit

You’re trying to format as COLS=48 but your font size is:

FS = 15
LEADING = 18


Even with Courier, 48 columns will overflow an 80mm receipt unless the font is very small. Result: either clipped text or forced scaling.

✅ Fix: For PDF receipts (browser share/print), use something like:

COLS = 42 (more realistic for PDF)

FS = 8 and LEADING = 10

(48 columns is fine for ESC/POS raw printing, but you’re doing PDF share printing.)

D) Wrap logic is inconsistent

You mix:

COLS and rule lines ("-"*COLS)

but wrap() uses simpleSplit() based on page width

Because your page width is wrong (165mm), simpleSplit() may not wrap at all, so customer names can exceed your intended width.

✅ Fix: Wrap by columns (textwrap) for consistency.

E) Small but important content issues

VAT: CY10353264O → last character is letter O, should be 0.

Your “BALANCE DUE” uses FONT_B, 24 (very large). On thermal PDF this can cause clipping/scaling.

✅ Fix VAT text + make “BALANCE DUE” bold but not huge (10–12pt max).

3) Drop-in replacement: fixed PDF builder (SPP-R310 friendly)

Put this in services/thermal_receipt_pdf.py:

from io import BytesIO
from decimal import Decimal
from reportlab.pdfgen import canvas
from reportlab.lib.units import mm
import textwrap

PDF_COLS = 42
FONT = "Courier"
FONT_B = "Courier-Bold"
FS = 8
LEADING = 10

PAGE_W = 80 * mm
LEFT = 4 * mm
RIGHT = 4 * mm
TOP = 4 * mm
BOT = 6 * mm

def money(v):
    try:
        return f"EUR {Decimal(str(v)):,.2f}"
    except Exception:
        return "EUR 0.00"

def _pad_right(label, value, width=PDF_COLS):
    label = str(label)
    value = str(value)
    gap = width - len(label) - len(value)
    if gap < 1:
        gap = 1
    return label + (" " * gap) + value

def _wrap_cols(s, width=PDF_COLS):
    s = (s or "").strip()
    if not s:
        return [""]
    return textwrap.wrap(s, width=width, break_long_words=True, replace_whitespace=False)

def build_delivery_receipt_pdf(data: dict) -> bytes:
    """
    Builds an 80mm thermal PDF optimized for mobile share/print to BIXOLON SPP-R310.
    """
    lines = []
    sep = "-" * PDF_COLS

    def add(s=""):
        lines.append(("L", s[:PDF_COLS]))

    def add_b(s):
        lines.append(("B", s[:PDF_COLS]))

    def add_c(s):
        lines.append(("C", s[:PDF_COLS]))

    def add_bc(s):
        lines.append(("BC", s[:PDF_COLS]))

    is_collected = bool(data.get("is_collected", False))
    is_credit = bool(data.get("is_credit", False))
    is_preview = bool(data.get("is_preview", False))
    is_amended = bool(data.get("is_amended", False))

    # Flags
    if is_preview:
        add_bc("*** PREVIEW - NOT A FINAL RECEIPT ***")
        add("")
    if is_amended:
        add_bc("*** AMENDED DELIVERY ***")
        add("")

    # Header
    add_bc("STEP EPLATTFORMA LTD")
    add_c("Digeni Akrita 13BC, 1055 Lefkosia")
    add_c("Tel: 7000 0394  VAT: CY103532640")
    add(sep)

    # Title / Status (content rules we discussed)
    if is_credit:
        add_bc("DELIVERY CONFIRMATION")
        add_bc("CREDIT ACCOUNT")
    elif is_collected:
        add_bc("PAYMENT RECEIPT")
        add_bc("STATUS: PAID")
    else:
        add_bc("DELIVERY CONFIRMATION / PAYMENT DUE")
        add_bc("STATUS: NOT COLLECTED")
    add(sep)

    # IDs (never wrap)
    receipt_no = str(data.get("receipt_no", "")).strip()
    date_str = str(data.get("date_str", "")).strip()
    add(_pad_right(f"Receipt: {receipt_no}", f"Date: {date_str}"))

    route_no = str(data.get("route_no", "")).strip()
    stop_no = str(data.get("stop_no", "")).strip()
    driver = str(data.get("driver_name", "")).strip()

    add(f"Route: {route_no}  Stop: {stop_no}")
    add(f"Driver: {driver}")

    cust_code = str(data.get("customer_code", "")).strip()
    if cust_code:
        add(f"Cust Code: {cust_code}")
    add(sep)

    # Customer
    add_b("CUSTOMER")
    for w in _wrap_cols(data.get("customer_name", "")):
        add(w)
    for w in _wrap_cols(data.get("customer_addr", "")):
        add(w)
    add(sep)

    # Invoices
    invoices = data.get("invoices") or []
    add_b(f"INVOICES ({len(invoices)})")
    for inv in invoices:
        inv_no = str(inv.get("invoice_no", "")).strip()
        inv_total = inv.get("total", None)
        if inv_total is not None:
            add(_pad_right(f"  {inv_no}", money(inv_total)))
        else:
            add(f"  {inv_no}"[:PDF_COLS])
    add(sep)

    # Amounts (always show expected/collected/balance)
    add_b("AMOUNTS")
    expected = Decimal(str(data.get("expected", "0") or "0"))
    collected = Decimal(str(data.get("collected", "0") or "0"))
    balance = Decimal(str(data.get("balance_due", "") or (expected - collected)))

    add(_pad_right("  Expected:", money(expected)))
    add(_pad_right("  Collected:", money(collected)))

    if is_collected:
        pm = str(data.get("payment_method", "")).upper().replace("_", " ").strip()
        add(_pad_right("  Method:", pm or "-"))

        if (data.get("payment_method") or "").lower() == "cheque":
            if data.get("cheque_number"):
                add(_pad_right("  Cheque No:", str(data["cheque_number"])))
            if data.get("cheque_date"):
                add(_pad_right("  Cheque Date:", str(data["cheque_date"])))

        if (data.get("payment_method") or "").lower() == "cash":
            if data.get("cash_received") is not None:
                add(_pad_right("  Cash Received:", money(data["cash_received"])))
            if data.get("change_given") is not None:
                add(_pad_right("  Change Given:", money(data["change_given"])))

    if is_credit:
        add_b("  ON ACCOUNT - NO BALANCE DUE")
    else:
        # Emphasize balance due when not collected or balance > 0
        if (not is_collected) or (balance > 0):
            add_b(_pad_right("  BALANCE DUE:", money(balance)))

    add(sep)

    # Notes
    notes = (data.get("notes") or "").strip()
    if notes:
        add_b("NOTES")
        for w in _wrap_cols(notes):
            add("  " + w if len(w) <= PDF_COLS - 2 else w[:PDF_COLS])
        add(sep)

    # Signatures (content rules we discussed)
    add("")
    add("Customer Signature (Delivery):")
    add("________________________________")
    add("")

    if is_collected:
        add("Customer Signature (Payment):")
        add("________________________________")
        add("")

    add("Driver Signature:")
    add("________________________________")

    # Dynamic height
    total_lines = len(lines)
    page_h = TOP + BOT + (total_lines * LEADING)

    buf = BytesIO()
    c = canvas.Canvas(buf, pagesize=(PAGE_W, page_h))
    y = page_h - TOP

    for style, txt in lines:
        if style == "C":
            c.setFont(FONT, FS)
            c.drawCentredString(PAGE_W / 2, y, txt)
        elif style == "B":
            c.setFont(FONT_B, FS)
            c.drawString(LEFT, y, txt)
        elif style == "BC":
            c.setFont(FONT_B, FS)
            c.drawCentredString(PAGE_W / 2, y, txt)
        else:
            c.setFont(FONT, FS)
            c.drawString(LEFT, y, txt)
        y -= LEADING

    c.showPage()
    c.save()
    buf.seek(0)
    return buf.getvalue()


This implements the content improvements we discussed:

No “payment collected” wording when NOT COLLECTED

Always shows Expected / Collected / Balance Due

Payment signature only when collected

Preview/Amended flags

Invoice list per line
