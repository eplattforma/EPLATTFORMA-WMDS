Below are Replit-ready build instructions (Flask + SQLAlchemy) to implement the Customer 360 analytics screen using only dw_* sales data + ps_customers + ps_items, including PY / Previous-period comparison and Item-RFM (stale items).

0) DB prerequisites (do this first)
A) Decide the DW “sales lines” source

This implementation assumes you have one DW table/view that represents sales lines with (at minimum):

customer_code_365

invoice_no_365

invoice_date (DATE or TIMESTAMP)

item_code_365

quantity

line_total_excl (or computable from fields you have)

If your table name differs, we’ll make it configurable via env var:

DW_SALES_LINE_TABLE (default: dw_transactions)

B) Add indexes (Postgres) for speed

Run these once in your DB (adjust table name if needed):

# ---- Helpers ---------------------------------------------------------------

def _parse_date(v: str) -> date | None:
    if not v:
        return None
    # expects YYYY-MM-DD
    return datetime.strptime(v, "%Y-%m-%d").date()

def _safe_shift_year(d: date, years: int) -> date:
    # handle Feb 29 -> Feb 28 in non-leap years
    try:
        return d.replace(year=d.year + years)
    except ValueError:
        # fallback for Feb 29
        return d.replace(month=2, day=28, year=d.year + years)

def _resolve_range(args) -> tuple[date, date]:
    """
    Priority:
      1) preset=last90|mtd|qtd|ytd|last30
      2) from=YYYY-MM-DD & to=YYYY-MM-DD
    Defaults to last90.
    """
    preset = (args.get("preset") or "").lower().strip()
    today = date.today()

    if preset == "last30":
        return today - timedelta(days=29), today
    if preset == "mtd":
        start = today.replace(day=1)
        return start, today
    if preset == "qtd":
        q = (today.month - 1) // 3 + 1
        start_month = 1 + (q - 1) * 3
        start = date(today.year, start_month, 1)
        return start, today
    if preset == "ytd":
        start = date(today.year, 1, 1)
        return start, today
    if preset == "last90" or preset == "":
        return today - timedelta(days=89), today

    # custom
    d_from = _parse_date(args.get("from"))
    d_to = _parse_date(args.get("to"))
    if not d_from or not d_to:
        # fallback if invalid
        return today - timedelta(days=89), today
    if d_to < d_from:
        d_from, d_to = d_to, d_from
    return d_from, d_to

def _compare_range(d_from: date, d_to: date, mode: str) -> tuple[date, date] | None:
    """
    mode:
      - prev: previous period of same length immediately preceding d_from
      - py: same dates one year earlier
    """
    mode = (mode or "").lower().strip()
    if mode not in ("prev", "py"):
        return None

    if mode == "py":
        return _safe_shift_year(d_from, -1), _safe_shift_year(d_to, -1)

    # prev
    length = (d_to - d_from).days + 1
    prev_end = d_from - timedelta(days=1)
    prev_start = prev_end - timedelta(days=length - 1)
    return prev_start, prev_end

def _role_ok() -> bool:
    # adjust roles to your system
    r = getattr(current_user, "role", None)
    return r in ("admin", "warehouse_manager", "manager", "accounting")

@bp.route("/api/search")
@login_required
def api_search_customers():
    if not _role_ok():
        return jsonify({"error": "forbidden"}), 403

    q = (request.args.get("q") or "").strip()
    if not q:
        return jsonify({"items": []})

    # Search both code and name
    sql = text("""
        SELECT customer_code_365 AS code,
               COALESCE(customer_name, customer_name_365, customer_name1, '') AS name
        FROM ps_customers
        WHERE customer_code_365 ILIKE :likeq
           OR COALESCE(customer_name, customer_name_365, customer_name1, '') ILIKE :likeq
        ORDER BY customer_code_365
        LIMIT 20
    """)
    likeq = f"%{q}%"
    rows = db.session.execute(sql, {"likeq": likeq}).mappings().all()
    return jsonify({"items": [dict(r) for r in rows]})

@bp.route("/api/<customer_code>/summary")
@login_required
def api_customer_summary(customer_code):
    if not _role_ok():
        return jsonify({"error": "forbidden"}), 403

    d_from, d_to = _resolve_range(request.args)
    compare = (request.args.get("compare") or "").lower().strip()  # prev|py|empty
    cmp_range = _compare_range(d_from, d_to, compare)

    # IMPORTANT: Adjust column names here if your DW differs.
    # We use line_total_excl and invoice_date.
    base_sql = f"""
        SELECT
            COALESCE(SUM(line_total_excl), 0) AS sales_excl,
            COALESCE(SUM(quantity), 0) AS qty,
            COUNT(DISTINCT invoice_no_365) AS invoices,
            COUNT(DISTINCT item_code_365) AS distinct_items
        FROM {DW_SALES_LINE_TABLE}
        WHERE customer_code_365 = :code
          AND invoice_date >= :d_from
          AND invoice_date <= :d_to
    """

    cur = db.session.execute(
        text(base_sql),
        {"code": customer_code, "d_from": d_from, "d_to": d_to}
    ).mappings().first()

    cur_sales = float(cur["sales_excl"] or 0)
    cur_invoices = int(cur["invoices"] or 0)
    avg_invoice = (cur_sales / cur_invoices) if cur_invoices else 0.0

    payload = {
        "range": {"from": str(d_from), "to": str(d_to)},
        "current": {
            "sales_excl": cur_sales,
            "qty": float(cur["qty"] or 0),
            "invoices": cur_invoices,
            "distinct_items": int(cur["distinct_items"] or 0),
            "avg_invoice": avg_invoice,
        },
        "compare_mode": compare if cmp_range else None,
        "compare": None,
        "delta": None
    }

    if cmp_range:
        c_from, c_to = cmp_range
        prev = db.session.execute(
            text(base_sql),
            {"code": customer_code, "d_from": c_from, "d_to": c_to}
        ).mappings().first()

        prev_sales = float(prev["sales_excl"] or 0)
        prev_invoices = int(prev["invoices"] or 0)
        prev_avg_invoice = (prev_sales / prev_invoices) if prev_invoices else 0.0

        payload["compare"] = {
            "range": {"from": str(c_from), "to": str(c_to)},
            "sales_excl": prev_sales,
            "qty": float(prev["qty"] or 0),
            "invoices": prev_invoices,
            "distinct_items": int(prev["distinct_items"] or 0),
            "avg_invoice": prev_avg_invoice,
        }

        def _delta(curv, prevv):
            dv = curv - prevv
            pct = (dv / prevv * 100.0) if prevv else None
            return {"value": dv, "pct": pct}

        payload["delta"] = {
            "sales_excl": _delta(cur_sales, prev_sales),
            "invoices": _delta(cur_invoices, prev_invoices),
            "avg_invoice": _delta(avg_invoice, prev_avg_invoice),
        }

    return jsonify(payload)

@bp.route("/api/<customer_code>/top-items")
@login_required
def api_customer_top_items(customer_code):
    if not _role_ok():
        return jsonify({"error": "forbidden"}), 403

    d_from, d_to = _resolve_range(request.args)
    limit = int(request.args.get("limit") or 20)
    limit = max(5, min(limit, 200))
    sort = (request.args.get("sort") or "sales").lower().strip()  # sales|qty

    order_by = "sales_excl DESC" if sort == "sales" else "qty DESC"

    sql = text(f"""
        SELECT
            t.item_code_365 AS item_code,
            COALESCE(i.item_name, i.item_name_365, i.item_description, '') AS item_name,
            COALESCE(SUM(t.quantity), 0) AS qty,
            COALESCE(SUM(t.line_total_excl), 0) AS sales_excl,
            COUNT(DISTINCT t.invoice_no_365) AS invoices
        FROM {DW_SALES_LINE_TABLE} t
        LEFT JOIN ps_items i ON i.item_code_365 = t.item_code_365
        WHERE t.customer_code_365 = :code
          AND t.invoice_date >= :d_from
          AND t.invoice_date <= :d_to
        GROUP BY t.item_code_365, COALESCE(i.item_name, i.item_name_365, i.item_description, '')
        ORDER BY {order_by}
        LIMIT :limit
    """)
    rows = db.session.execute(sql, {
        "code": customer_code, "d_from": d_from, "d_to": d_to, "limit": limit
    }).mappings().all()

    return jsonify({
        "range": {"from": str(d_from), "to": str(d_to)},
        "items": [dict(r) for r in rows]
    })

@bp.route("/api/<customer_code>/invoices")
@login_required
def api_customer_invoices(customer_code):
    if not _role_ok():
        return jsonify({"error": "forbidden"}), 403

    d_from, d_to = _resolve_range(request.args)
    page = int(request.args.get("page") or 1)
    page_size = int(request.args.get("page_size") or 25)
    page = max(1, page)
    page_size = max(10, min(page_size, 100))
    offset = (page - 1) * page_size

    # grouped from sales lines
    sql = text(f"""
        SELECT
            invoice_no_365 AS invoice_no,
            DATE(invoice_date) AS invoice_date,
            COALESCE(SUM(line_total_excl), 0) AS sales_excl,
            COALESCE(SUM(quantity), 0) AS qty,
            COUNT(DISTINCT item_code_365) AS distinct_items
        FROM {DW_SALES_LINE_TABLE}
        WHERE customer_code_365 = :code
          AND invoice_date >= :d_from
          AND invoice_date <= :d_to
        GROUP BY invoice_no_365, DATE(invoice_date)
        ORDER BY DATE(invoice_date) DESC, invoice_no_365 DESC
        LIMIT :limit OFFSET :offset
    """)

    rows = db.session.execute(sql, {
        "code": customer_code, "d_from": d_from, "d_to": d_to,
        "limit": page_size, "offset": offset
    }).mappings().all()

    return jsonify({
        "range": {"from": str(d_from), "to": str(d_to)},
        "page": page,
        "page_size": page_size,
        "items": [dict(r) for r in rows]
    })

@bp.route("/api/<customer_code>/item-rfm")
@login_required
def api_customer_item_rfm(customer_code):
    if not _role_ok():
        return jsonify({"error": "forbidden"}), 403

    lookback_days = int(request.args.get("lookback_days") or 730)  # 2 years default
    stale_days = int(request.args.get("stale_days") or 90)
    min_freq = int(request.args.get("min_freq") or 1)
    stale_only = (request.args.get("stale_only") or "1") == "1"

    today = date.today()
    d_from = today - timedelta(days=max(30, min(lookback_days, 3650)))

    # Item-RFM per customer:
    # Recency: days since last purchase
    # Frequency: distinct invoices
    # Monetary: sum(line_total_excl)
    # IMPORTANT: adjust invoice_date/line_total_excl if your DW differs.
    sql = text(f"""
        WITH agg AS (
            SELECT
                t.item_code_365 AS item_code,
                MAX(DATE(t.invoice_date)) AS last_purchase_date,
                COUNT(DISTINCT t.invoice_no_365) AS frequency,
                COALESCE(SUM(t.line_total_excl), 0) AS monetary
            FROM {DW_SALES_LINE_TABLE} t
            WHERE t.customer_code_365 = :code
              AND t.invoice_date >= :d_from
              AND t.invoice_date <= :d_to
            GROUP BY t.item_code_365
        )
        SELECT
            a.item_code,
            COALESCE(i.item_name, i.item_name_365, i.item_description, '') AS item_name,
            a.last_purchase_date,
            (DATE(:d_to) - a.last_purchase_date) AS recency_days,
            a.frequency,
            a.monetary
        FROM agg a
        LEFT JOIN ps_items i ON i.item_code_365 = a.item_code
        WHERE a.frequency >= :min_freq
          AND (:stale_only = 0 OR (DATE(:d_to) - a.last_purchase_date) >= :stale_days)
        ORDER BY recency_days DESC, monetary DESC
        LIMIT 500
    """)

    rows = db.session.execute(sql, {
        "code": customer_code,
        "d_from": d_from,
        "d_to": today,
        "stale_days": stale_days,
        "min_freq": min_freq,
        "stale_only": 1 if stale_only else 0
    }).mappings().all()

    return jsonify({
        "lookback_days": lookback_days,
        "stale_days": stale_days,
        "min_freq": min_freq,
        "stale_only": stale_only,
        "items": [dict(r) for r in rows]
    })

  let selectedCustomerCode = null;

  function money(v) {
    const n = Number(v || 0);
    return n.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});
  }
  function num(v) {
    const n = Number(v || 0);
    return n.toLocaleString();

    if (v === null || v === undefined) return "—";

    if (!preset && to) p.set("to", to);
