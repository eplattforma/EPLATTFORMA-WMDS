C) Parsing utility (strict)

Create services/delivery_days.py:

import re
from datetime import datetime

VALID_DOW = set(range(1, 8))
VALID_WEEK = {1, 2}

def parse_delivery_days_strict(raw: str):
    # Extract all 2-digit tokens from messy strings (handles trailing commas, dots, spaces)
    tokens = re.findall(r"\d{2}", raw or "")
    if not tokens:
        return {"status": "EMPTY", "slots": [], "invalid": []}

    slots = []
    invalid = []

    for t in tokens:
        dow = int(t[0])
        wk = int(t[1])
        if dow not in VALID_DOW or wk not in VALID_WEEK:
            invalid.append(t)
        else:
            slots.append((dow, wk))

    # If ANY invalid exists => whole value is INVALID (per your requirement)
    if invalid:
        return {"status": "INVALID", "slots": [], "invalid": sorted(set(invalid))}

    # Deduplicate
    slots = sorted(set(slots))
    return {"status": "OK", "slots": slots, "invalid": []}

from services.delivery_days import parse_delivery_days_strict
from models import PSCustomer, CustomerDeliverySlot
from app import db
from datetime import datetime
import json

def refresh_customer_delivery_slots(customer: PSCustomer):
    parsed = parse_delivery_days_strict(customer.delivery_days)

    customer.delivery_days_status = parsed["status"]
    customer.delivery_days_invalid_tokens = json.dumps(parsed["invalid"]) if parsed["invalid"] else None
    customer.delivery_days_parsed_at = datetime.utcnow()

    # Always clear old slots
    CustomerDeliverySlot.query.filter_by(customer_code_365=customer.customer_code_365).delete()

    if parsed["status"] == "OK":
        for dow, wk in parsed["slots"]:
            db.session.add(CustomerDeliverySlot(
                customer_code_365=customer.customer_code_365,
                dow=dow,
                week_code=wk
            ))

2) Wed “any week” = must contain BOTH 31 AND 32

This is “dow=3 AND week_code in (1,2)” + HAVING distinct week_code count = 2

from sqlalchemy import func

q = (
  db.session.query(PSCustomer)
  .join(CustomerDeliverySlot, PSCustomer.customer_code_365 == CustomerDeliverySlot.customer_code_365)
  .filter(PSCustomer.delivery_days_status == "OK")
  .filter(CustomerDeliverySlot.dow == 3, CustomerDeliverySlot.week_code.in_([1,2]))
  .group_by(PSCustomer.customer_code_365)
  .having(func.count(func.distinct(CustomerDeliverySlot.week_code)) == 2)
)


That guarantees the customer has both 31 and 32.
