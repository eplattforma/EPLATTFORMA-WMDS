2.1 Step open helper + step completion UI
function openStep(which) {
  const map = {
    exceptions: 'cExceptions',
    signature: 'cSignature',
    payment: 'cPayment',
    print: 'cPrint'
  };
  const id = map[which];
  if (!id) return;
  const el = document.getElementById(id);
  if (!el) return;
  const collapse = bootstrap.Collapse.getOrCreateInstance(el);
  collapse.show();
}

function setStepDone(stepId, done) {
  const el = document.getElementById(stepId);
  const mark = document.getElementById(stepId + '-mark');
  if (!el || !mark) return;
  el.classList.remove('done', 'todo');
  el.classList.add(done ? 'done' : 'todo');
  mark.textContent = done ? '✅' : '⬜';
}

function isPostDatedCheque() {
  const methodEl = document.getElementById('cod-method');
  const method = methodEl ? methodEl.value : '';
  if (method !== 'cheque') return false;

  const chequeDateInput = document.getElementById('cheque-date');
  if (!chequeDateInput || !chequeDateInput.value) return false;

  const d = new Date(chequeDateInput.value + 'T00:00:00');
  const today = new Date(); today.setHours(0,0,0,0);
  return d > today;
}

function isSignatureRequired() {
  if (isCredit) return true;
  const methodEl = document.getElementById('cod-method');
  const method = methodEl ? methodEl.value : '';
  if (method === 'online') return true;
  if (method === 'cheque' && isPostDatedCheque()) return true;
  return false;
}

function updateSignatureRequirementUI() {
  const required = isSignatureRequired();
  const sw = document.getElementById('pod-physical-switch');
  const badge = document.getElementById('sig-required-badge');
  const help = document.getElementById('sig-help');
  const warn = document.getElementById('sig-warn');
  const block = document.getElementById('sigBlock');

  if (!sw || !badge || !help || !warn || !block) return;

  if (required) {
    badge.textContent = 'REQUIRED';
    badge.className = 'badge bg-danger ms-2';
    help.textContent = 'Required for Credit / Pay Online / Post-dated cheque.';
    warn.style.display = sw.checked ? 'none' : 'block';
    block.classList.toggle('border-danger', !sw.checked);
  } else {
    badge.textContent = 'Optional';
    badge.className = 'badge bg-secondary ms-2';
    help.textContent = 'Recommended, but optional for Cash/Day Cheque/Card.';
    warn.style.display = 'none';
    block.classList.remove('border-danger');
  }
}

function getBalanceDue() {
  if (isCredit) return 0;
  const invoiceTotal = totalInvoicesAmount;
  const exceptionsTotal = exceptions.reduce((sum, ex) => sum + (ex.exception_value || 0), 0);

  const methodEl = document.getElementById('cod-method');
  const method = methodEl ? methodEl.value : 'cash';
  const receivedInput = document.getElementById('cod-received');
  const received = (method === 'online') ? 0 : (parseFloat(receivedInput?.value) || 0);

  return (invoiceTotal - exceptionsTotal - received);
}

function refreshHeaderAndSteps() {
  const inv = totalInvoicesAmount;
  const exTot = exceptions.reduce((sum, ex) => sum + (ex.exception_value || 0), 0);
  const entered = (!isCredit && (document.getElementById('cod-method')?.value !== 'online'))
    ? (parseFloat(document.getElementById('cod-received')?.value) || 0) : 0;

  const due = Math.max(0, inv - exTot); // amount to collect ignoring "entered"
  document.getElementById('hdr-inv').textContent = inv.toFixed(2);
  document.getElementById('hdr-ex').textContent = exTot.toFixed(2);
  document.getElementById('hdr-entered').textContent = entered.toFixed(2);
  document.getElementById('hdr-collect').textContent = due.toFixed(2);

  // Exceptions step done
  const exDone = (exceptions.length === 0) || window.exceptionsProofPrinted === true || window.exceptionsSkipAllowed === true;
  setStepDone('st-ex', exDone);

  // Signature step done
  const sigRequired = isSignatureRequired();
  const sigChecked = !!document.getElementById('pod-physical-switch')?.checked;
  setStepDone('st-sig', (!sigRequired) || sigChecked);

  // Payment step done
  let payDone = true;
  if (!isCredit) {
    const m = document.getElementById('cod-method')?.value || '';
    payDone = !!m;
    // if cash/card/cheque and due>0 => must have received >0 OR note
    const dueNow = Math.max(0, inv - exTot);
    const received = (m === 'online') ? 0 : (parseFloat(document.getElementById('cod-received')?.value) || 0);
    if ((m === 'cash' || m === 'card' || m === 'cheque') && dueNow > 0 && received <= 0) payDone = false;
  }
  setStepDone('st-pay', payDone);

  // Print step done
  const printDone = isCredit ? true : (window.paymentDocPrinted === true);
  const exPrintDone = (exceptions.length === 0) ? true : (window.exceptionsProofPrinted === true || window.exceptionsSkipAllowed === true);
  setStepDone('st-prn', printDone && exPrintDone);

  // Enable/disable finish
  const blockers = [];
  if (!exDone) blockers.push("Print Exceptions Proof (or skip with note).");
  if (sigRequired && !sigChecked) blockers.push("Signed invoice required.");
  if (!payDone) blockers.push("Payment entry incomplete.");
  if (!printDone) blockers.push("Payment document must be printed.");
  if (!exPrintDone) blockers.push("Exceptions proof required.");

  const btnFinish = document.getElementById('btnFinishStop');
  const reason = document.getElementById('finishReason');
  if (btnFinish) btnFinish.disabled = blockers.length > 0;
  if (reason) {
    reason.style.display = blockers.length ? 'block' : 'none';
    reason.textContent = blockers.length ? ("To close stop: " + blockers.join(" ")) : "";
  }

  // Exceptions proof button
  const btnEx = document.getElementById('btnPrintExceptions');
  if (btnEx) btnEx.disabled = exceptions.length === 0;
  const exSummary = document.getElementById('ex-summary');
  if (exSummary) exSummary.textContent = `Exceptions: ${exceptions.length} (−€${exTot.toFixed(2)})`;

  // Step 4 requirement badges
  const reqEx = document.getElementById('reqExProof');
  if (reqEx) reqEx.textContent = (exceptions.length > 0) ? "Required" : "Optional";
}

async function printPaymentDoc() {
