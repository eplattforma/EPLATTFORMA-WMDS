let rawCents = ""; // digits only, "1234" => 12.34

function setMethod(m) {
  const sel = document.getElementById('cod-method');
  if (sel) sel.value = m;
  togglePaymentControls();
  calculateCODVariance();
}

function _setCollectedFromRaw() {
  const cents = rawCents === "" ? 0 : parseInt(rawCents, 10);
  const amount = (cents / 100).toFixed(2);

  document.getElementById('cod-received').value = amount;
  document.getElementById('cod-display').textContent = `€ ${amount}`;
  calculateCODVariance();
}

function kp(d) {
  rawCents = (rawCents + d).replace(/^0+/, ""); // trim leading zeros
  _setCollectedFromRaw();
}
function kpBack() {
  rawCents = rawCents.slice(0, -1);
  _setCollectedFromRaw();
}
function kpClear() {
  rawCents = "";
  _setCollectedFromRaw();
}
function kpExact() {
  const invoiceTotal = totalInvoicesAmount;
  const exceptionsTotal = exceptions.reduce((sum, ex) => sum + (ex.exception_value || 0), 0);
  const due = Math.max(0, invoiceTotal - exceptionsTotal);
  rawCents = String(Math.round(due * 100));
  _setCollectedFromRaw();
}
