# Task #32 — Route Batch for Normal Items

## What & Why

FIX-002 gives normal batches hard locking via `locked_by_batch_id`.
This task adds automatic batch accumulation for normal items — the same
auto-accumulation that cooler batches already have for SENSITIVE items.

When any invoice is attached to a route, the system automatically:
1. Extracts SENSITIVE items → `COOLER-ROUTE-<id>` session (existing)
2. Extracts normal items → `ROUTE-BATCH-<id>` session (new)

Both sessions stay **open** (accumulating) until the warehouse manager
explicitly locks them. Only after locking can picking be assigned.

This eliminates the need to manually create standard batches for route
items. The route IS the batch.

## Prerequisites

- FIX-002 must be deployed and verified first
- `summer_cooler_mode_enabled` flag must be ON for cooler extraction
- New flag `route_batch_mode_enabled` controls this feature (default OFF)

## Done Looks Like

- When `route_batch_mode_enabled = true`, attaching any invoice to a
  route automatically creates or updates `ROUTE-BATCH-<route_id>`
  with all non-SENSITIVE, unpicked items from that invoice locked via
  `locked_by_batch_id`
- The batch is visible in `/admin/batch/manage` with type badge "Route"
- The route detail page shows a "Route Batch Status" card alongside the
  existing "Cooler Status" card, with a "Lock Route Batch" button
- After locking, the batch is assignable to a picker exactly like any
  standard batch — same picking interface, same location sorting
- Late-joining invoices auto-accumulate into the same session until
  the session is locked
- After locking, new invoices joining the route create a new
  `ROUTE-BATCH-<id>-2` sibling (same pattern as cooler)
- All existing manual batch creation continues to work unchanged —
  `route_batch_mode_enabled = false` preserves today's behaviour exactly

### S.1 — New `_extract_normal_items_for_route()` function

Create in `services/cooler_route_extraction.py` alongside the existing
cooler extraction function. Follow the identical pattern:

```python
ROUTE_BATCH_SESSION_TYPE = "route_batch"

def _get_or_create_route_batch_session(route_id: int,
                                        creator: str) -> dict:
    """
    Return the active route_batch session for route_id,
    creating one if none exists.
    Naming: ROUTE-BATCH-<id>, ROUTE-BATCH-<id>-2, etc.
    Same sibling pattern as cooler sessions.
    """
    base_name = f"ROUTE-BATCH-{route_id}"
    existing = db.session.execute(text("""
        SELECT id, name, status
        FROM batch_picking_sessions
        WHERE name LIKE :pattern
          AND session_type = 'route_batch'
          AND status NOT IN ('Completed', 'Cancelled', 'Archived')
        ORDER BY id DESC LIMIT 1
    """), {"pattern": f"{base_name}%"}).fetchone()

    if existing:
        return {"id": existing[0], "name": existing[1], "created": False}

    # Create new session
    next_name = _next_route_batch_name(route_id)  # same as cooler sibling logic
    db.session.execute(text("""
        INSERT INTO batch_picking_sessions
          (name, batch_number, zones, picking_mode, session_type,
           route_id, status, created_by, created_at)
        VALUES
          (:name, :bn, :zones, 'Consolidated', 'route_batch',
           :rid, 'Created', :user, NOW())
    """), {
        "name": next_name,
        "bn": _generate_batch_number(),  # existing helper
        "zones": "MAIN",  # updated as items are added
        "rid": route_id,
        "user": creator,
    })
    new_id = db.session.execute(
        text("SELECT id FROM batch_picking_sessions WHERE name = :n"),
        {"n": next_name}
    ).scalar()
    return {"id": new_id, "name": next_name, "created": True}


def extract_normal_items_for_route_stop_invoices(rsi_list: list,
                                                  creator: str) -> dict:
    """
    Extract non-SENSITIVE items from each RouteStopInvoice into the
    route's ROUTE-BATCH session. Idempotent.

    Called from services/__init__.py:attach_invoices_to_stop()
    immediately after extract_sensitive_for_route_stop_invoices().

    Returns: {extracted, already_present, skipped_picked}
    """
    from models import Setting
    flag = Setting.get(db.session, "route_batch_mode_enabled", "false")
    if flag.lower() != "true":
        return {"extracted": 0, "already_present": 0, "skipped_picked": 0}

    extracted = 0
    already_present = 0
    skipped_picked = 0

    # Group RSIs by route_id
    route_groups = {}
    for rsi in rsi_list:
        route_id = _get_route_id_for_stop(rsi.route_stop_id)
        if route_id:
            route_groups.setdefault(route_id, []).append(rsi)

    for route_id, rsil in route_groups.items():
        session = _get_or_create_route_batch_session(route_id, creator)
        session_id = session["id"]

        for rsi in rsil:
            invoice_no = rsi.invoice_no
            if _session_is_locked(session_id):
                # Session is locked — late addition, create sibling
                session = _get_or_create_route_batch_session(
                    route_id, creator
                )
                session_id = session["id"]

            # Get non-SENSITIVE items for this invoice
            items = db.session.execute(text("""
                SELECT ii.id, ii.item_code, ii.zone, ii.is_picked,
                       ii.locked_by_batch_id, ii.pick_status
                FROM invoice_items ii
                LEFT JOIN ps_items_dw psi
                       ON psi.item_code_365 = ii.item_code
                WHERE ii.invoice_no = :inv
                  AND ii.is_picked = false
                  AND (psi.wms_zone IS NULL
                       OR psi.wms_zone NOT IN ('SENSITIVE'))
            """), {"inv": invoice_no}).fetchall()

            for item in items:
                item_id = item[0]
                current_lock = item[4]

                if current_lock == session_id:
                    already_present += 1
                    continue
                if current_lock is not None:
                    # Locked by another batch (cooler or manual)
                    continue
                if item[3]:  # is_picked
                    skipped_picked += 1
                    continue

                # Lock to this route batch
                db.session.execute(text("""
                    UPDATE invoice_items
                    SET locked_by_batch_id = :sid
                    WHERE id = :iid
                      AND locked_by_batch_id IS NULL
                """), {"sid": session_id, "iid": item_id})

                # Add BatchSessionInvoice row if not present
                db.session.execute(text("""
                    INSERT INTO batch_session_invoices
                      (batch_session_id, invoice_no)
                    VALUES (:sid, :inv)
                    ON CONFLICT DO NOTHING
                """), {"sid": session_id, "inv": invoice_no})

                extracted += 1

    db.session.commit()
    return {
        "extracted": extracted,
        "already_present": already_present,
        "skipped_picked": skipped_picked,
    }
```

### S.3 — `get_grouped_items()`: route_batch branch

In `models.py`, add a new branch for `session_type == 'route_batch'`
in `get_grouped_items()`. This is identical to the Consolidated branch
but uses `locked_by_batch_id == self.id` as the primary filter:

```python
elif getattr(self, 'session_type', None) == 'route_batch':
    # Route batches use hard lock — no zone filter needed
    # (FIX-002 ensures all items are locked by this session)
    conds = [InvoiceItem.locked_by_batch_id == self.id]
    if not include_picked:
        conds.extend([
            InvoiceItem.is_picked == False,
            InvoiceItem.pick_status.in_(
                ['not_picked', 'reset', 'skipped_pending']
            ),
        ])
    all_items = db.session.query(InvoiceItem).filter(
        and_(*conds)
    ).all()

    # Exclude any cooler items that might have slipped through
    all_items = [
        it for it in all_items
        if (it.invoice_no, it.item_code) not in cooler_keys
    ]

    all_items = sort_items_for_picking(all_items)  # location order

    grouped = {}
    for item in all_items:
        key = (item.item_code, item.location or "")
        if key not in grouped:
            grouped[key] = {
                'item_code': item.item_code,
                'location': item.location,
                'barcode': item.barcode,
                'zone': item.zone,
                'item_name': item.item_name,
                'unit_type': item.unit_type,
                'pack': item.pack,
                'total_qty': 0,
                'source_items': [],
            }
        qty = item.qty or 0
        grouped[key]['total_qty'] += qty
        grouped[key]['source_items'].append({
            'invoice_no': item.invoice_no,
            'item_code': item.item_code,
            'qty': qty,
            'id': f"{item.invoice_no}-{item.item_code}",
        })
    return sort_batch_items(list(grouped.values()))
```

### S.5 — Lock route batch endpoint

```python
@batch_bp.route("/route-batch/<int:route_id>/lock", methods=["POST"])
@login_required
@require_permission("picking.manage_batches")
def lock_route_batch(route_id):
    """Lock a route batch so it becomes assignable."""
    session = BatchPickingSession.query.filter(
        BatchPickingSession.route_id == route_id,
        BatchPickingSession.session_type == 'route_batch',
        BatchPickingSession.status.notin_(
            ['Completed', 'Cancelled', 'Archived']
        ),
    ).first()
    if not session:
        flash("No active route batch found for this route.", "warning")
        return redirect(request.referrer or url_for('admin_dashboard'))

    session.sequence_locked_at = get_utc_now()
    session.sequence_locked_by = current_user.username
    db.session.commit()

    flash(
        f"Route batch {session.name} locked. "
        "Assign to a picker from Manage Batches.",
        "success"
    )
    return redirect(request.referrer or url_for('admin_dashboard'))
```
