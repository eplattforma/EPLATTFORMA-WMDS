def parse_date(s: str) -> date:

def period_prev(start: date, end: date):
    days = (end - start).days + 1
    prev_end = start - timedelta(days=1)
    prev_start = prev_end - timedelta(days=days - 1)
    return prev_start, prev_end

    # Same period last year (simple -365; OK operationally)
    return start - timedelta(days=365), end - timedelta(days=365)

@bp.route("/api/kpis")
@login_required
def kpis():
    customer_code = request.args.get("customer_code", "").strip()
    group_name = request.args.get("group_name", "").strip()
    start = parse_date(request.args["start"])
    end = parse_date(request.args["end"])
    compare_mode = request.args.get("compare_mode", "prev")  # prev|py|none

    if not customer_code:
        return jsonify({"error": "customer_code required"}), 400

    if not group_name:
        group_name = get_customer_group(customer_code) or ""

    comp_start = comp_end = None
    if compare_mode == "prev":
        comp_start, comp_end = period_prev(start, end)
    elif compare_mode == "py":
        comp_start, comp_end = period_py(start, end)

    sql = text("""
    WITH
    params AS (
      SELECT :start::date AS start_date, :end::date AS end_date
    ),
    base AS (
      SELECT
        h.invoice_date::date AS invoice_date,
        h.customer_code,
        h.invoice_no_365,
        l.item_code_365,
        CASE
          WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN -ABS(COALESCE(l.quantity,0))
          ELSE COALESCE(l.quantity,0)
        END AS qty_net,
        CASE
          WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN -ABS(COALESCE(l.line_total_excl,0))
          ELSE COALESCE(l.line_total_excl,0)
        END AS sales_excl_net,
        CASE
          WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN 0
          ELSE GREATEST(COALESCE(l.quantity,0),0)
        END AS qty_gross,
        CASE
          WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN 0
          ELSE GREATEST(COALESCE(l.line_total_excl,0),0)
        END AS sales_excl_gross
      FROM dw_invoice_header h
      JOIN dw_invoice_line l
        ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    ),
    cust AS (
      SELECT
        SUM(sales_excl_net) AS net_sales,
        SUM(qty_net)        AS net_qty,
        COUNT(DISTINCT invoice_no_365) AS orders,
        COUNT(*) AS lines,
        COUNT(DISTINCT item_code_365) AS uniq_items,
        SUM(sales_excl_gross) AS gross_sales,
        NULLIF(SUM(qty_gross),0) AS gross_qty
      FROM base
      WHERE customer_code = :cust
    ),
    peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_base AS (
      SELECT b.*
      FROM base b
      JOIN peerset p ON p.customer_code = b.customer_code
    ),
    peer AS (
      SELECT
        SUM(sales_excl_net) AS net_sales,
        SUM(qty_net)        AS net_qty,
        COUNT(DISTINCT invoice_no_365) AS orders,
        COUNT(*) AS lines,
        COUNT(DISTINCT item_code_365) AS uniq_items,
        SUM(sales_excl_gross) AS gross_sales,
        NULLIF(SUM(qty_gross),0) AS gross_qty,
        (SELECT COUNT(*) FROM peerset) AS peer_customers
      FROM peer_base
    )
    SELECT
      (SELECT net_sales FROM cust) AS cust_net_sales,
      (SELECT net_qty FROM cust) AS cust_net_qty,
      (SELECT orders FROM cust) AS cust_orders,
      (SELECT lines FROM cust) AS cust_lines,
      (SELECT uniq_items FROM cust) AS cust_uniq_items,
      (SELECT gross_sales FROM cust) AS cust_gross_sales,
      (SELECT gross_qty FROM cust) AS cust_gross_qty,

      (SELECT net_sales FROM peer) AS peer_net_sales,
      (SELECT net_qty FROM peer) AS peer_net_qty,
      (SELECT orders FROM peer) AS peer_orders,
      (SELECT lines FROM peer) AS peer_lines,
      (SELECT uniq_items FROM peer) AS peer_uniq_items,
      (SELECT gross_sales FROM peer) AS peer_gross_sales,
      (SELECT gross_qty FROM peer) AS peer_gross_qty,
      (SELECT peer_customers FROM peer) AS peer_customers
    """)
    row = db.session.execute(sql, {
        "start": start, "end": end,
        "cust": customer_code,
        "grp": group_name
    }).mappings().first()

    if not row:
        return jsonify({"error": "no data"}), 404

    def safe_div(a, b):
        return float(a) / float(b) if a is not None and b not in (None, 0) else None

    # KPIs
    cust_aov = safe_div(row["cust_net_sales"], row["cust_orders"])
    peer_aov = safe_div(row["peer_net_sales"], row["peer_orders"])
    peer_avg_sales_per_customer = safe_div(row["peer_net_sales"], row["peer_customers"])

    cust_unit_price_gross = safe_div(row["cust_gross_sales"], row["cust_gross_qty"])
    peer_unit_price_gross = safe_div(row["peer_gross_sales"], row["peer_gross_qty"])

    out = {
      "customer_code": customer_code,
      "group_name": group_name,
      "period": {"start": str(start), "end": str(end), "compare_mode": compare_mode},
      "customer": {
        "net_sales": float(row["cust_net_sales"] or 0),
        "orders": int(row["cust_orders"] or 0),
        "aov": cust_aov,
        "uniq_items": int(row["cust_uniq_items"] or 0),
        "unit_price_gross": cust_unit_price_gross
      },
      "peers": {
        "peer_customers": int(row["peer_customers"] or 0),
        "net_sales_total": float(row["peer_net_sales"] or 0),
        "avg_sales_per_customer": peer_avg_sales_per_customer,
        "aov": peer_aov,
        "unit_price_gross": peer_unit_price_gross
      },
      "indexes": {
        "sales_index_vs_peer_avg": safe_div(float(row["cust_net_sales"] or 0), peer_avg_sales_per_customer),
        "price_index_gross": safe_div(cust_unit_price_gross, peer_unit_price_gross)
      }
    }

    # optional compare period block
    if compare_mode in ("prev", "py"):
        cs, ce = comp_start, comp_end
        out["compare_period"] = {"start": str(cs), "end": str(ce)}

    return jsonify(out)

@login_required
def not_bought():
    customer_code = request.args.get("customer_code", "").strip()
    group_name = request.args.get("group_name", "").strip()
    start = parse_date(request.args["start"])
    end = parse_date(request.args["end"])
    penetration_min = float(request.args.get("penetration_min", "0.30"))
    limit = int(request.args.get("limit", "200"))

    if not group_name:
        group_name = get_customer_group(customer_code) or ""

    sql = text("""
    WITH base AS (
      SELECT
        h.customer_code,
        l.item_code_365,
        CASE WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN -ABS(COALESCE(l.quantity,0)) ELSE COALESCE(l.quantity,0) END AS qty_net,
        CASE WHEN COALESCE(h.invoice_type,'') ILIKE '%RETURN%' THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
    ),
    peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (
      SELECT COUNT(*) AS n FROM peerset
    ),
    peer_item AS (
      SELECT
        b.item_code_365,
        COUNT(DISTINCT b.customer_code) FILTER (WHERE b.qty_net > 0) AS buyers,
        SUM(b.sales_net) AS peer_sales
      FROM base b
      JOIN peerset p ON p.customer_code = b.customer_code
      GROUP BY b.item_code_365
    ),
    cust_item AS (
      SELECT item_code_365, SUM(qty_net) AS cust_qty
      FROM base
      WHERE customer_code = :cust
      GROUP BY item_code_365
    )
    SELECT
      pi.item_code_365,
      i.item_name,
      i.category,
      i.brand,
      (pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) AS penetration,
      (pi.peer_sales::float / NULLIF((SELECT n FROM peer_cnt),0)) AS peer_avg_sales_per_customer,
      COALESCE(ci.cust_qty,0) AS cust_qty,
      ((pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) * (pi.peer_sales::float / NULLIF((SELECT n FROM peer_cnt),0))) AS opportunity_score
    FROM peer_item pi
    LEFT JOIN cust_item ci ON ci.item_code_365 = pi.item_code_365
    LEFT JOIN ps_items_dw i ON i.item_code_365 = pi.item_code_365
    WHERE COALESCE(ci.cust_qty,0) = 0
      AND (pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) >= :pen
    ORDER BY opportunity_score DESC NULLS LAST
    LIMIT :lim
    """)
    rows = db.session.execute(sql, {
      "start": start, "end": end,
      "cust": customer_code, "grp": group_name,
      "pen": penetration_min, "lim": limit
    }).mappings().all()
    return jsonify({"rows": [dict(r) for r in rows]})

    if compare_mode == "prev":
        comp_start, comp_end = period_prev(start, end)
    elif compare_mode == "py":
        comp_start, comp_end = period_py(start, end)

    # load both windows in one query (curr + comp)
    min_date = comp_start if comp_start and comp_start < start else start
    max_date = end

    sql = text(f"""
    WITH base AS (
      SELECT
        h.invoice_date::date AS d,
        h.customer_code,
        h.invoice_no_365,
        l.item_code_365,

        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.quantity,0)) ELSE COALESCE(l.quantity,0) END AS qty_net,
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net,

        CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.quantity,0),0) END AS qty_gross,
        CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.line_total_excl,0),0) END AS sales_gross

      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN :min_date::date AND :max_date::date
    ),
    tagged AS (
      SELECT *,
        CASE
          WHEN d BETWEEN :start::date AND :end::date THEN 'curr'
          WHEN :has_comp = 1 AND d BETWEEN :comp_start::date AND :comp_end::date THEN 'comp'
          ELSE NULL
        END AS period_tag
      FROM base
    ),
    peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (
      SELECT COUNT(*) AS n FROM peerset
    ),
    agg AS (
      SELECT
        period_tag,

        -- customer
        SUM(sales_net) FILTER (WHERE customer_code = :cust) AS cust_net_sales,
        COUNT(DISTINCT invoice_no_365) FILTER (WHERE customer_code = :cust) AS cust_orders,
        COUNT(DISTINCT item_code_365) FILTER (WHERE customer_code = :cust) AS cust_uniq_items,
        SUM(sales_gross) FILTER (WHERE customer_code = :cust) AS cust_gross_sales,
        NULLIF(SUM(qty_gross) FILTER (WHERE customer_code = :cust),0) AS cust_gross_qty,

        -- peers total
        SUM(sales_net) FILTER (WHERE customer_code IN (SELECT customer_code FROM peerset)) AS peer_net_sales_total,
        COUNT(DISTINCT invoice_no_365) FILTER (WHERE customer_code IN (SELECT customer_code FROM peerset)) AS peer_orders_total,
        SUM(sales_gross) FILTER (WHERE customer_code IN (SELECT customer_code FROM peerset)) AS peer_gross_sales_total,
        NULLIF(SUM(qty_gross) FILTER (WHERE customer_code IN (SELECT customer_code FROM peerset)),0) AS peer_gross_qty_total

      FROM tagged
      WHERE period_tag IS NOT NULL
      GROUP BY period_tag
    )
    SELECT
      (SELECT n FROM peer_cnt) AS peer_customers,
      jsonb_build_object(
        'cust_net_sales', COALESCE((SELECT cust_net_sales FROM agg WHERE period_tag='curr'),0),
        'cust_orders', COALESCE((SELECT cust_orders FROM agg WHERE period_tag='curr'),0),
        'cust_uniq_items', COALESCE((SELECT cust_uniq_items FROM agg WHERE period_tag='curr'),0),
        'cust_gross_sales', COALESCE((SELECT cust_gross_sales FROM agg WHERE period_tag='curr'),0),
        'cust_gross_qty', (SELECT cust_gross_qty FROM agg WHERE period_tag='curr'),
        'peer_net_sales_total', COALESCE((SELECT peer_net_sales_total FROM agg WHERE period_tag='curr'),0),
        'peer_orders_total', COALESCE((SELECT peer_orders_total FROM agg WHERE period_tag='curr'),0),
        'peer_gross_sales_total', COALESCE((SELECT peer_gross_sales_total FROM agg WHERE period_tag='curr'),0),
        'peer_gross_qty_total', (SELECT peer_gross_qty_total FROM agg WHERE period_tag='curr')
      ) AS curr,
      CASE WHEN :has_comp = 1 THEN jsonb_build_object(
        'cust_net_sales', COALESCE((SELECT cust_net_sales FROM agg WHERE period_tag='comp'),0),
        'cust_orders', COALESCE((SELECT cust_orders FROM agg WHERE period_tag='comp'),0),
        'cust_uniq_items', COALESCE((SELECT cust_uniq_items FROM agg WHERE period_tag='comp'),0)
      ) ELSE NULL END AS comp
    ;
    """)

    has_comp = 1 if (compare_mode in ("prev", "py")) else 0

    row = db.session.execute(sql, {
      "cust": cust,
      "grp": grp,
      "start": start,
      "end": end,
      "has_comp": has_comp,
      "comp_start": comp_start or start,
      "comp_end": comp_end or start,
      "min_date": min_date,
      "max_date": max_date,
    }).mappings().first()

    if not row:
        return jsonify({"error": "no data"}), 404

    peer_customers = int(row["peer_customers"] or 0)
    curr = row["curr"] or {}
    comp = row["comp"]

    def safe_div(a, b):
        return (float(a) / float(b)) if a is not None and b not in (None, 0) else None

    cust_net_sales = float(curr.get("cust_net_sales", 0))
    cust_orders = int(curr.get("cust_orders", 0))
    cust_uniq_items = int(curr.get("cust_uniq_items", 0))

    peer_net_sales_total = float(curr.get("peer_net_sales_total", 0))
    peer_avg_sales_per_customer = safe_div(peer_net_sales_total, peer_customers)

    cust_aov = safe_div(cust_net_sales, cust_orders)
    peer_aov = safe_div(peer_net_sales_total, int(curr.get("peer_orders_total", 0)))

    cust_unit_price_gross = safe_div(float(curr.get("cust_gross_sales", 0)), curr.get("cust_gross_qty"))
    peer_unit_price_gross = safe_div(float(curr.get("peer_gross_sales_total", 0)), curr.get("peer_gross_qty_total"))

    out = {
      "customer_code": cust,
      "group_name": grp,
      "period": {"start": str(start), "end": str(end), "compare_mode": compare_mode},
      "customer": {
        "net_sales": cust_net_sales,
        "orders": cust_orders,
        "aov": cust_aov,
        "uniq_items": cust_uniq_items,
        "unit_price_gross": cust_unit_price_gross
      },
      "peers": {
        "peer_customers": peer_customers,
        "net_sales_total": peer_net_sales_total,
        "avg_sales_per_customer": peer_avg_sales_per_customer,
        "aov": peer_aov,
        "unit_price_gross": peer_unit_price_gross
      },
      "indexes": {
        "sales_index_vs_peer_avg": safe_div(cust_net_sales, peer_avg_sales_per_customer),
        "price_index_gross": safe_div(cust_unit_price_gross, peer_unit_price_gross),
      }
    }

    if comp is not None:
        comp_sales = float(comp.get("cust_net_sales", 0))
        comp_orders = int(comp.get("cust_orders", 0))
        out["compare_period"] = {"start": str(comp_start), "end": str(comp_end)}
        out["deltas"] = {
          "net_sales_abs": cust_net_sales - comp_sales,
          "net_sales_pct": safe_div((cust_net_sales - comp_sales), comp_sales),
          "orders_abs": cust_orders - comp_orders,
          "orders_pct": safe_div((cust_orders - comp_orders), comp_orders),
        }

    return jsonify(out)

def not_bought():

    end = parse_date(request.args["end"])

    limit = int(request.args.get("limit", "200"))

    if not grp:
        grp = get_customer_group(cust) or ""

    sql = text(f"""
    WITH base AS (
      SELECT
        h.customer_code,
        l.item_code_365,
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.quantity,0)) ELSE COALESCE(l.quantity,0) END AS qty_net,
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
    ),
    peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (SELECT COUNT(*) AS n FROM peerset),
    peer_item AS (
      SELECT
        b.item_code_365,
        COUNT(DISTINCT b.customer_code) FILTER (WHERE b.qty_net > 0) AS buyers,
        SUM(b.sales_net) FILTER (WHERE b.qty_net > 0) AS peer_sales
      FROM base b
      JOIN peerset p ON p.customer_code = b.customer_code
      GROUP BY b.item_code_365
    ),
    cust_item AS (
      SELECT item_code_365, SUM(qty_net) AS cust_qty
      FROM base
      WHERE customer_code = :cust
      GROUP BY item_code_365
    )
    SELECT
      pi.item_code_365,
      i.item_name,
      i.category,
      i.brand,
      (pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) AS penetration,
      (pi.peer_sales::float / NULLIF((SELECT n FROM peer_cnt),0)) AS peer_avg_sales_per_customer,
      COALESCE(ci.cust_qty,0) AS cust_qty,
      ((pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) * (pi.peer_sales::float / NULLIF((SELECT n FROM peer_cnt),0))) AS opportunity_score
    FROM peer_item pi
    LEFT JOIN cust_item ci ON ci.item_code_365 = pi.item_code_365
    LEFT JOIN ps_items_dw i ON i.item_code_365 = pi.item_code_365
    WHERE COALESCE(ci.cust_qty,0) = 0
      AND (pi.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) >= :pen
    ORDER BY opportunity_score DESC NULLS LAST
    LIMIT :lim
    """)
    rows = db.session.execute(sql, {
      "start": start, "end": end,
      "cust": cust, "grp": grp,
      "pen": penetration_min, "lim": limit
    }).mappings().all()

    return jsonify({"rows": [dict(r) for r in rows]})

             SUM(CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.quantity,0),0) END) AS comp_qty_gross,
             SUM(CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.line_total_excl,0),0) END) AS comp_sales_gross

    ),

      SELECT l.item_code_365,
             SUM(CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.quantity,0),0) END) AS curr_qty_gross

    ),

      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (SELECT COUNT(*) AS n FROM peerset),
    peer_curr_buyers AS (
      SELECT l.item_code_365,
             COUNT(DISTINCT h.customer_code) FILTER (WHERE NOT ({RETURN_PREDICATE}) AND COALESCE(l.quantity,0) > 0) AS buyers
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      JOIN peerset p ON p.customer_code = h.customer_code
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
      GROUP BY l.item_code_365

    SELECT
      c.item_code_365,

    LEFT JOIN ps_items_dw i ON i.item_code_365 = c.item_code_365

    WITH base AS (
      SELECT
        h.customer_code,
        COALESCE(i.category,'(Uncategorized)') AS category,
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      LEFT JOIN ps_items_dw i ON i.item_code_365 = l.item_code_365
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
    ),
    peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    cust_cat AS (
      SELECT category, SUM(sales_net) AS sales
      FROM base
      WHERE customer_code = :cust
      GROUP BY category
    ),
    peer_cat AS (
      SELECT category, SUM(sales_net) AS sales
      FROM base
      WHERE customer_code IN (SELECT customer_code FROM peerset)
      GROUP BY category
    ),
    totals AS (
      SELECT
        (SELECT SUM(sales) FROM cust_cat) AS cust_total,
        (SELECT SUM(sales) FROM peer_cat) AS peer_total
    )
    SELECT
      COALESCE(c.category, p.category) AS category,
      COALESCE(c.sales,0) AS cust_sales,
      COALESCE(p.sales,0) AS peer_sales,
      COALESCE(c.sales,0) / NULLIF((SELECT cust_total FROM totals),0) AS cust_share,
      COALESCE(p.sales,0) / NULLIF((SELECT peer_total FROM totals),0) AS peer_share,
      (COALESCE(c.sales,0) / NULLIF((SELECT cust_total FROM totals),0))
        -
      (COALESCE(p.sales,0) / NULLIF((SELECT peer_total FROM totals),0)) AS share_diff
    FROM cust_cat c
    FULL OUTER JOIN peer_cat p ON p.category = c.category
    ORDER BY ABS(
      (COALESCE(c.sales,0) / NULLIF((SELECT cust_total FROM totals),0))
      -
      (COALESCE(p.sales,0) / NULLIF((SELECT peer_total FROM totals),0))
    ) DESC NULLS LAST
    LIMIT :lim
    """)
    rows = db.session.execute(sql, {
      "cust": cust, "grp": grp,
      "start": start, "end": end,
      "lim": limit

@login_required
def price_outliers():
    denied = require_role()
    if denied: return denied

    cust = (request.args.get("customer_code") or "").strip()
    grp = (request.args.get("group_name") or "").strip()

    end = parse_date(request.args["end"])

    WITH peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    base_gross AS (
      SELECT
        h.customer_code,
        l.item_code_365,
        SUM(CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.quantity,0),0) END) AS qty_gross,
        SUM(CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.line_total_excl,0),0) END) AS sales_gross
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365

      GROUP BY h.customer_code, l.item_code_365
    ),

      SELECT item_code_365,
             SUM(qty_gross) AS qty,
             SUM(sales_gross) AS sales
      FROM base_gross
      WHERE customer_code = :cust
      GROUP BY item_code_365
      HAVING SUM(qty_gross) >= :min_qty
    ),
    peer_item AS (
      SELECT item_code_365,
             SUM(qty_gross) AS qty,
             SUM(sales_gross) AS sales
      FROM base_gross
      WHERE customer_code IN (SELECT customer_code FROM peerset)
      GROUP BY item_code_365
      HAVING SUM(qty_gross) >= :min_qty
    )
    SELECT
      c.item_code_365,
      i.item_name,
      i.category,
      i.brand,
      (c.sales / NULLIF(c.qty,0)) AS cust_unit_price,
      (p.sales / NULLIF(p.qty,0)) AS peer_unit_price,
      (c.sales / NULLIF(c.qty,0)) / NULLIF((p.sales / NULLIF(p.qty,0)),0) AS price_index,
      c.qty AS cust_qty,
      p.qty AS peer_qty
    FROM cust_item c
    JOIN peer_item p ON p.item_code_365 = c.item_code_365
    LEFT JOIN ps_items_dw i ON i.item_code_365 = c.item_code_365
    ORDER BY ABS(((c.sales / NULLIF(c.qty,0)) / NULLIF((p.sales / NULLIF(p.qty,0)),0)) - 1) DESC NULLS LAST
    LIMIT :lim
    """)
    rows = db.session.execute(sql, {
      "cust": cust, "grp": grp,
      "start": start, "end": end,
      "min_qty": min_qty,
      "lim": limit
    }).mappings().all()

    return jsonify({"rows": [dict(r) for r in rows]})

@bp.route("/api/trends-monthly")
@login_required
def trends_monthly():
    denied = require_role()
    if denied: return denied

    cust = (request.args.get("customer_code") or "").strip()
    grp = (request.args.get("group_name") or "").strip()
    start = parse_date(request.args["start"])
    end = parse_date(request.args["end"])

    if not cust:
        return jsonify({"error": "customer_code required"}), 400

    if not grp:
        grp = get_customer_group(cust) or ""

    sql = text(f"""
    WITH peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (SELECT COUNT(*)::int AS n FROM peerset),
    base AS (
      SELECT
        date_trunc('month', h.invoice_date)::date AS month_start,
        h.customer_code,
        h.invoice_no_365,
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net,
        CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE 1 END AS is_purchase_line
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
    ),
    cust_m AS (
      SELECT
        month_start,
        SUM(sales_net) AS cust_net_sales,
        COUNT(DISTINCT invoice_no_365) FILTER (WHERE is_purchase_line=1) AS cust_orders
      FROM base
      WHERE customer_code = :cust
      GROUP BY month_start
    ),
    peer_m AS (
      SELECT
        month_start,
        SUM(sales_net) AS peer_net_sales_total,
        COUNT(DISTINCT invoice_no_365) FILTER (WHERE is_purchase_line=1) AS peer_orders_total
      FROM base
      WHERE customer_code IN (SELECT customer_code FROM peerset)
      GROUP BY month_start
    )
    SELECT
      COALESCE(c.month_start, p.month_start) AS month_start,
      COALESCE(c.cust_net_sales,0) AS cust_net_sales,
      COALESCE(c.cust_orders,0) AS cust_orders,
      COALESCE(p.peer_net_sales_total,0) AS peer_net_sales_total,
      COALESCE(p.peer_orders_total,0) AS peer_orders_total,
      (COALESCE(p.peer_net_sales_total,0) / NULLIF((SELECT n FROM peer_cnt),0)) AS peer_avg_sales_per_customer,
      (COALESCE(p.peer_orders_total,0)::float / NULLIF((SELECT n FROM peer_cnt),0)) AS peer_avg_orders_per_customer,
      (SELECT n FROM peer_cnt) AS peer_customers
    FROM cust_m c
    FULL OUTER JOIN peer_m p ON p.month_start = c.month_start
    ORDER BY month_start
    """)

    rows = db.session.execute(sql, {
      "cust": cust, "grp": grp, "start": start, "end": end
    }).mappings().all()

    return jsonify({"rows": [dict(r) for r in rows]})

Recency is measured vs the selected end date (so the screen is consistent).

“Last purchase date” is computed from non-return purchases (gross logic).

Monetary is net sales (returns reduce it).

@bp.route("/api/item-rfm")
@login_required
def item_rfm():
    denied = require_role()
    if denied: return denied

    cust = (request.args.get("customer_code") or "").strip()
    grp = (request.args.get("group_name") or "").strip()
    end = parse_date(request.args["end"])
    lookback_days = int(request.args.get("lookback_days", "365"))
    limit = int(request.args.get("limit", "300"))

    if not cust:
        return jsonify({"error": "customer_code required"}), 400

    if not grp:
        grp = get_customer_group(cust) or ""

    start = end - timedelta(days=lookback_days - 1)

    sql = text(f"""
    WITH peerset AS (
      SELECT customer_code
      FROM customer_reporting_group_members
      WHERE group_name = :grp
        AND customer_code <> :cust
    ),
    peer_cnt AS (SELECT COUNT(*)::int AS n FROM peerset),

    base AS (
      SELECT
        h.customer_code,
        h.invoice_no_365,
        h.invoice_date::date AS d,
        l.item_code_365,

        -- net monetary (returns reduce)
        CASE WHEN {RETURN_PREDICATE} THEN -ABS(COALESCE(l.line_total_excl,0)) ELSE COALESCE(l.line_total_excl,0) END AS sales_net,

        -- gross purchase qty: returns excluded
        CASE WHEN {RETURN_PREDICATE} THEN 0 ELSE GREATEST(COALESCE(l.quantity,0),0) END AS qty_gross
      FROM dw_invoice_header h
      JOIN dw_invoice_line l ON l.invoice_no_365 = h.invoice_no_365
      WHERE h.invoice_date::date BETWEEN :start::date AND :end::date
    ),

    cust_item AS (
      SELECT
        item_code_365,
        MAX(d) FILTER (WHERE qty_gross > 0) AS last_purchase_date,
        COUNT(DISTINCT invoice_no_365) FILTER (WHERE qty_gross > 0) AS purchase_orders,
        SUM(qty_gross) AS qty_gross,
        SUM(sales_net) AS sales_net
      FROM base
      WHERE customer_code = :cust
      GROUP BY item_code_365
    ),

    peer_buyers AS (
      SELECT
        item_code_365,
        COUNT(DISTINCT customer_code) FILTER (WHERE qty_gross > 0) AS buyers
      FROM base
      WHERE customer_code IN (SELECT customer_code FROM peerset)
      GROUP BY item_code_365
    )

    SELECT
      ci.item_code_365,
      i.item_name,
      i.category,
      i.brand,
      ci.last_purchase_date,
      (EXTRACT(day FROM (:end::date - ci.last_purchase_date)))::int AS days_since_last_purchase,
      ci.purchase_orders,
      ci.qty_gross,
      ci.sales_net,
      (ci.sales_net / NULLIF(ci.purchase_orders,0)) AS avg_sales_per_order,
      (ci.sales_net / NULLIF(ci.qty_gross,0)) AS avg_unit_price,
      (pb.buyers::float / NULLIF((SELECT n FROM peer_cnt),0)) AS peer_penetration
    FROM cust_item ci
    LEFT JOIN peer_buyers pb ON pb.item_code_365 = ci.item_code_365
    LEFT JOIN ps_items_dw i ON i.item_code_365 = ci.item_code_365
    ORDER BY days_since_last_purchase DESC NULLS LAST, ci.sales_net DESC NULLS LAST
    LIMIT :lim
    """)

    rows = db.session.execute(sql, {
      "cust": cust, "grp": grp, "start": start, "end": end, "lim": limit
    }).mappings().all()

    return jsonify({
      "lookback": {"start": str(start), "end": str(end), "days": lookback_days},
      "rows": [dict(r) for r in rows]
    })

function sparklineSVG(seriesA, seriesB, w=520, h=90){
  // seriesA: customer, seriesB: peer avg
  const pad = 6;
  const all = [...seriesA, ...seriesB].filter(x => x !== null && x !== undefined);
  if(!all.length) return "";

  const min = Math.min(...all);
  const max = Math.max(...all);
  const den = (max-min) || 1;

  function pts(series){
    return series.map((v,i)=>{
      const x = pad + (i*(w-2*pad)/(Math.max(series.length-1,1)));
      const y = h - pad - ((v-min)/den)*(h-2*pad);
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    }).join(" ");
  }

  const pA = pts(seriesA.map(v=>v??min));
  const pB = pts(seriesB.map(v=>v??min));

  return `
  <svg width="${w}" height="${h}" viewBox="0 0 ${w} ${h}" style="border:1px solid #ddd;border-radius:10px;">
    <polyline fill="none" stroke="black" stroke-width="2" points="${pA}" />
    <polyline fill="none" stroke="#888" stroke-width="2" stroke-dasharray="4 3" points="${pB}" />
    <text x="10" y="16" font-size="12" fill="#000">Customer</text>
    <text x="90" y="16" font-size="12" fill="#888">Peers avg</text>
  </svg>`;
}
