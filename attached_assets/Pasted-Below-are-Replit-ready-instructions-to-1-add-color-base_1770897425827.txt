2) Add JS helper to map value → class

File: static/js/customer_pricing.js

Add:

function signClass(n, eps=0.005) {
  const v = Number(n || 0);
  if (Math.abs(v) <= eps) return "zero";
  return v > 0 ? "pos" : "neg";
}
function kpiCardClass(n) {
  const s = signClass(n);
  return s === "pos" ? "kpi-pos" : (s === "neg" ? "kpi-neg" : "kpi-zero");
}
