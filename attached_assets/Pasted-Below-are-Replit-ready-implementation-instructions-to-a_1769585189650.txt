@reports_bp.route("/reserved-stock-777/settings/save", methods=["POST"])
@login_required
def reserved_stock_777_save_settings():
    payload = request.get_json(force=True) or {}
    season_code = (payload.get("season_code_365") or "").strip()
    if not season_code:
        return jsonify({"success": False, "error": "Missing season_code_365"}), 400

    s = SeasonSupplierSetting.query.get(season_code)
    if not s:
        s = SeasonSupplierSetting(season_code_365=season_code)
        db.session.add(s)

    s.supplier_code = (payload.get("supplier_code") or "").strip() or None
    s.email_to = (payload.get("email_to") or "").strip() or None
    s.email_cc = (payload.get("email_cc") or "").strip() or None
    s.email_comment = (payload.get("email_comment") or "").strip() or None

    db.session.commit()
    return jsonify({"success": True})

2) Send PO endpoint (PS365 + Email)

Add route:

@reports_bp.route("/reserved-stock-777/send-po", methods=["POST"])
@login_required
def reserved_stock_777_send_po():
    payload = request.get_json(force=True) or {}
    season_code = (payload.get("season_code_365") or "").strip()
    if not season_code:
        return jsonify({"success": False, "error": "Missing season_code_365"}), 400

    # Load settings
    setting = SeasonSupplierSetting.query.get(season_code)
    if not setting or not setting.supplier_code or not setting.email_to:
        return jsonify({"success": False, "error": "Missing supplier_code or email_to in settings for this season"}), 400

    # Load report data (re-use your existing cached JSON OR DB snapshot)
    # Preferred: read the same JSON your report already uses.
    report = load_reserved_stock_777_report()  # implement helper: reads reports_cache/reserved_stock_777_latest.json
    season_rows = [r for r in report["rows"] if (r.get("season_code_365") or "").strip() == season_code]

    # Only items that need ordering:
    # recommended rule: reserved > ordered_po_777
    lines = []
    for r in season_rows:
        reserved = Decimal(str(r.get("reserved_777") or "0"))
        ordered = Decimal(str(r.get("ordered_po_777") or "0"))
        need = reserved - ordered
        if need > 0:
            lines.append({
                "item_code_365": r["item_code_365"],
                "qty": float(need)  # or int if required by PS365
            })

    if not lines:
        return jsonify({"success": False, "error": "No items require ordering (reserved <= ordered)."}), 400

    # 1) Create PO in PS365
    # Use your existing ps365_client/call_ps365 logic
    po_no = create_po_in_ps365(
        supplier_code=setting.supplier_code,
        store_code="777",
        lines=lines,
        season_code=season_code
    )

    # 2) Send email to supplier
    send_supplier_email(
        to=setting.email_to,
        cc=setting.email_cc,
        subject=f"Purchase Order {po_no} - Season {season_code}",
        html_body=render_po_email_html(po_no, season_code, lines, setting.email_comment),
    )

    # 3) Force refresh of report snapshot (optional but recommended)
    # run script or rebuild cache so the ordered_po_777 updates on next load
    trigger_reserved_stock_777_refresh()

    return jsonify({"success": True, "po_no": po_no})

<script>
async function saveSeasonSettings(seasonCode) {
  const supplier = document.querySelector(`#supplier_${seasonCode}`).value;
  const emailTo = document.querySelector(`#emailto_${seasonCode}`).value;
  const emailCc = document.querySelector(`#emailcc_${seasonCode}`).value;
  const comment = document.querySelector(`#comment_${seasonCode}`).value;

  const res = await fetch('/reports/reserved-stock-777/settings/save', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      season_code_365: seasonCode,
      supplier_code: supplier,
      email_to: emailTo,
      email_cc: emailCc,
      email_comment: comment
    })
  });
  const data = await res.json();
  if (!data.success) alert(data.error || "Save failed");
  else alert("Saved");
}

async function sendPO(seasonCode) {
  if (!confirm("Create PO in PS365 and email supplier?")) return;

  const res = await fetch('/reports/reserved-stock-777/send-po', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({season_code_365: seasonCode})
  });
  const data = await res.json();
  if (!data.success) {
    alert(data.error || "Send PO failed");
    return;
  }
  alert("PO created: " + data.po_no);
  window.location.reload();
}
</script>
