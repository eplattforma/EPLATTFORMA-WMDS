def assign_invoice_to_stop(invoice_no, route_stop_id, actor):
    # validate invoice exists, stop exists
    stop = RouteStop.query.get(route_stop_id)
    inv = Invoice.query.get(invoice_no)

    # close any active mapping (should not exist for new assignment; safe anyway)
    active = RouteStopInvoice.query.filter_by(invoice_no=invoice_no, is_active=True).first()
    if active:
        active.is_active = False
        active.effective_to = utcnow()
        active.changed_by = actor

    rsi = RouteStopInvoice(
        route_stop_id=route_stop_id,
        invoice_no=invoice_no,
        status="PENDING",
        is_active=True,
        effective_from=utcnow(),
        changed_by=actor
    )
    db.session.add(rsi)

    # optional cache
    inv.stop_id = route_stop_id
    inv.route_id = stop.shipment_id

    db.session.commit()

Pseudo:

def reroute_invoice(invoice_no, new_route_stop_id, actor, reason=None, notes=None):
    now = utcnow()

    old = RouteStopInvoice.query.filter_by(invoice_no=invoice_no, is_active=True).first()
    if not old:
        raise ValueError("Invoice has no active stop mapping")

    old_stop_id = old.route_stop_id

    # close old mapping
    old.is_active = False
    old.effective_to = now
    old.changed_by = actor

    # new mapping
    new = RouteStopInvoice(
        route_stop_id=new_route_stop_id,
        invoice_no=invoice_no,
        status=old.status or "PENDING",
        is_active=True,
        effective_from=now,
        changed_by=actor
    )
    db.session.add(new)

    # history
    db.session.add(InvoiceRouteHistory(
        invoice_no=invoice_no,
        route_id=None,          # optional; or stop.shipment_id
        route_stop_id=new_route_stop_id,
        action="REROUTED",
        reason=reason,
        notes=f"from_stop={old_stop_id}; {notes or ''}",
        actor_username=actor
    ))

    # optional cache sync
    inv = Invoice.query.get(invoice_no)
    stop = RouteStop.query.get(new_route_stop_id)
    inv.stop_id = new_route_stop_id
    inv.route_id = stop.shipment_id

    db.session.commit()

D2) Stop status = aggregate of invoices under stop

Compute stop status dynamically:

If all active invoices under stop are DELIVERED → stop = DELIVERED

If all FAILED → stop = FAILED

If mix → stop = PARTIAL

If none finalized → stop = IN_PROGRESS

Implement as a query in code or create a DB view.

Example SQL for stop aggregate:

SELECT
  rsi.route_stop_id,
  CASE
    WHEN SUM(CASE WHEN rsi.status='DELIVERED' THEN 1 ELSE 0 END) = COUNT(*) THEN 'DELIVERED'
    WHEN SUM(CASE WHEN rsi.status='FAILED' THEN 1 ELSE 0 END) = COUNT(*) THEN 'FAILED'
    WHEN SUM(CASE WHEN rsi.status IN ('DELIVERED','FAILED') THEN 1 ELSE 0 END) > 0 THEN 'PARTIAL'
    ELSE 'IN_PROGRESS'
  END AS stop_status
FROM route_stop_invoice rsi
WHERE rsi.is_active = true
GROUP BY rsi.route_stop_id;

WITH inv AS (
  SELECT rsi.invoice_no, rsi.status

cash AS (
  SELECT
    COALESCE(SUM(expected_amount),0) expected,
    COALESCE(SUM(received_amount),0) received,
    COALESCE(SUM(CASE WHEN ps365_synced_at IS NULL THEN 1 ELSE 0 END),0) not_synced_count
  FROM cod_receipts
  WHERE route_id = :shipment_id
)
SELECT
  (SELECT COUNT(*) FROM inv) AS invoices_total,
  (SELECT COUNT(*) FROM inv WHERE status='DELIVERED') AS delivered,
  (SELECT COUNT(*) FROM inv WHERE status='FAILED') AS failed,
  (SELECT COUNT(*) FROM inv WHERE status='PARTIAL') AS partial,

  (SELECT cnt FROM open_cases) AS open_cases,
  (SELECT cnt FROM unresolved_disc) AS unresolved_discrepancies,
  (SELECT expected FROM cash) AS cash_expected,
  (SELECT received FROM cash) AS cash_received,
  (SELECT not_synced_count FROM cash) AS receipts_not_synced;

Recommended small table (adds high value)

Create invoice_payment_expectations:

invoice_no (PK/FK)

expected_payment_method (CASH/CHEQUE/CARD/BANK_TRANSFER/CREDIT)

is_cod boolean

expected_amount numeric (default invoice total)

snapshot fields: customer_code_365, terms_code, due_days, captured_at

Populate it when invoices are imported:

from PS365 payment/terms if available

otherwise infer from credit_terms (customer_code -> allow_cash, allow_cheque, etc.)

Then your “missing COD” becomes deterministic.
