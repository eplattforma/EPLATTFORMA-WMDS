3) Create utility functions: next delivery + ordering window open (week parity + weekends excluded)

Create file: services/crm_order_window.py

import pytz
from datetime import date, datetime, time, timedelta

ATHENS_TZ = pytz.timezone("Europe/Athens")

DOW_MAP = {
    "MON": 0, "TUE": 1, "WED": 2, "THU": 3, "FRI": 4, "SAT": 5, "SUN": 6
}

def iso_week_parity(d: date) -> int:
    # returns 1 for odd ISO week, 2 for even ISO week
    wk = d.isocalendar().week
    return 2 if (wk % 2 == 0) else 1

def next_delivery_date_for_slot(dow: str, week_code: str, from_date: date | None = None) -> date:
    """
    Find the next date >= from_date with weekday == dow and ISO week parity == week_code (1/2).
    """
    if not from_date:
        from_date = date.today()

    target_wd = DOW_MAP[dow.upper()]
    target_parity = int(week_code)

    # Start checking from today forward (up to 28 days is enough)
    for i in range(0, 28):
        d = from_date + timedelta(days=i)
        if d.weekday() == target_wd and iso_week_parity(d) == target_parity:
            return d

    # Fallback (should never hit)
    return from_date + timedelta(days=14)

def subtract_working_minutes(dt_local: datetime, minutes: int) -> datetime:
    """
    Subtract N minutes from dt_local counting only Mon-Fri minutes (exclude Sat/Sun completely).
    """
    remaining = minutes
    cur = dt_local
    step = timedelta(minutes=1)

    while remaining > 0:
        cur = cur - step
        if cur.weekday() in (5, 6):  # Sat=5, Sun=6
            continue
        remaining -= 1

    return cur

def order_window_open_at(delivery_date: date, window_hours: int, anchor_time_str: str = "00:01") -> datetime:
    """
    delivery_date is local date.
    anchor_time_str default 00:01 local.
    Window open time = delivery_anchor_dt_local - window_hours (excluding weekends).
    """
    hh, mm = anchor_time_str.split(":")
    delivery_anchor_local = ATHENS_TZ.localize(datetime.combine(delivery_date, time(int(hh), int(mm))))
    open_local = subtract_working_minutes(delivery_anchor_local, window_hours * 60)
    return open_local

def is_window_open(now_local: datetime, open_local: datetime, delivery_anchor_local: datetime) -> bool:
    return open_local <= now_local < delivery_anchor_local
