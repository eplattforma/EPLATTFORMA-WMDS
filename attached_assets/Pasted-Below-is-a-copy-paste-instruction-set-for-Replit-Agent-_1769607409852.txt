1) Add a canonical delivery status helper (new file)
Create: delivery_status.py
# delivery_status.py
from typing import Optional

STATUS_MAP = {
    "OUT_FOR_DELIVERY": "out_for_delivery",
    "DELIVERED": "delivered",
    "FAILED": "delivery_failed",
    "DELIVERY_FAILED": "delivery_failed",
    "RETURNED": "returned_to_warehouse",
    "RETURNED_TO_WAREHOUSE": "returned_to_warehouse",
    "SHIPPED": "shipped",
}

TERMINAL_DELIVERY_STATUSES = {"delivered", "delivery_failed", "returned_to_warehouse"}

def normalize_status(v: Optional[str]) -> Optional[str]:
    if v is None:
        return None
    s = str(v).strip()
    if not s:
        return None
    if s in STATUS_MAP:
        return STATUS_MAP[s]
    up = s.upper()
    if up in STATUS_MAP:
        return STATUS_MAP[up]
    low = s.lower()
    if low == "returned":
        return "returned_to_warehouse"
    return low
