## 1. `services/cooler_box_planner.py` — Add pre-pick estimate function

At the very end of the file (after the `generate_box_plan` function), append this new function:

```python
def pre_pick_estimate(route_id, delivery_date):
    """
    Return a volume/box estimate for a route BEFORE picking starts.
    Reads ALL cooler queue rows (pending + picked + exception).
    Called by the route_list view to show managers what boxes to prepare.
    """
    from sqlalchemy import text
    from app import db

    _EMPTY = {
        "total_items": 0, "total_volume_l": 0,
        "missing_dimension_items": [], "stops": [],
        "box_plan": [], "recommended_box_type": None,
        "all_dims_present": True,
    }

    # Fetch all cooler queue rows for this route (any status)
    rows = db.session.execute(text("""
        SELECT cq.item_code, cq.quantity, cq.stop_no, cq.customer_name
        FROM cooler_queue cq
        WHERE cq.route_id = :r AND cq.delivery_date = :d
        ORDER BY cq.stop_no
    """), {"r": route_id, "d": delivery_date}).fetchall()

    if not rows:
        return _EMPTY

    # Collect unique item codes for bulk dimension fetch
    item_codes = list({r[0] for r in rows})
    dim_rows = db.session.execute(text("""
        SELECT item_code,
               COALESCE(length_cm, 0), COALESCE(width_cm, 0), COALESCE(height_cm, 0)
        FROM ps_items_dw
        WHERE item_code = ANY(:codes)
    """), {"codes": item_codes}).fetchall()
    dims = {r[0]: (float(r[1]), float(r[2]), float(r[3])) for r in dim_rows}

    # Fetch active box types ordered smallest to largest
    box_types = db.session.execute(text("""
        SELECT id, name, internal_volume_cm3, fill_efficiency
        FROM cooler_box_types
        WHERE is_active = true
        ORDER BY internal_volume_cm3
    """)).fetchall()
    box_types = [
        {"id": r[0], "name": r[1],
         "capacity_cm3": float(r[2]) * float(r[3])}
        for r in box_types
    ]

    if not box_types:
        return _EMPTY

    # Group items by stop (LIFO: highest stop first)
    from collections import defaultdict
    stop_map = defaultdict(list)
    missing = []
    for item_code, qty, stop_no, customer_name in rows:
        l, w, h = dims.get(item_code, (0, 0, 0))
        vol = l * w * h * int(qty or 1)
        if l == 0 or w == 0 or h == 0:
            missing.append({"item_code": item_code})
            vol = 0
        stop_map[stop_no or 0].append({
            "item_code": item_code, "quantity": qty,
            "customer_name": customer_name, "volume_cm3": vol,
        })

    sorted_stops = sorted(stop_map.keys(), reverse=True)  # LIFO

    # Build stop summary list
    stop_summary = []
    for stop_no in sorted(stop_map.keys()):
        items = stop_map[stop_no]
        vol_l = round(sum(i["volume_cm3"] for i in items) / 1000, 2)
        missing_count = sum(1 for i in items
                            if dims.get(i["item_code"], (0,0,0))[0] == 0)
        stop_summary.append({
            "stop_no": stop_no,
            "customer_names": list({i["customer_name"] for i in items if i["customer_name"]}),
            "item_count": sum(int(i["quantity"] or 1) for i in items),
            "volume_l": vol_l,
            "missing_dims": missing_count,
        })

    # Simulate box plan: group stops into boxes using largest box type
    largest_box = box_types[-1]
    box_plan = []
    current_vol = 0
    current_stops = []
    current_missing = 0

    for stop_no in sorted_stops:
        items = stop_map[stop_no]
        stop_vol = sum(i["volume_cm3"] for i in items)
        stop_missing = sum(1 for i in items
                           if dims.get(i["item_code"], (0,0,0))[0] == 0)

        if current_stops and (current_vol + stop_vol) > largest_box["capacity_cm3"]:
            box_plan.append((current_stops[:], current_vol, current_missing))
            current_stops, current_vol, current_missing = [], 0, 0

        current_stops.append(stop_no)
        current_vol += stop_vol
        current_missing += stop_missing

    if current_stops:
        box_plan.append((current_stops, current_vol, current_missing))

    # For each planned box, find the smallest box type that fits
    rendered_boxes = []
    for stop_list, vol_cm3, miss in box_plan:
        fitting = next((b for b in box_types if b["capacity_cm3"] >= vol_cm3), None)
        chosen = fitting or box_types[-1]
        over = vol_cm3 > chosen["capacity_cm3"]
        fill_pct = round(vol_cm3 / chosen["capacity_cm3"] * 100) if chosen["capacity_cm3"] else 0
        stops_sorted = sorted(stop_list)
        stops_display = (f"Stop {stops_sorted[0]}" if len(stops_sorted) == 1
                         else f"Stops {stops_sorted[-1]}→{stops_sorted[0]}")
        rendered_boxes.append({
            "box_type_name": chosen["name"],
            "stops_display": stops_display,
            "estimated_volume_l": round(vol_cm3 / 1000, 2),
            "estimated_fill_pct": fill_pct,
            "over_capacity": over,
            "missing_dims": miss,
        })

    total_vol = sum(
        sum(i["volume_cm3"] for i in items)
        for items in stop_map.values()
    )

    # Recommended default box: smallest that fits the largest single stop
    max_stop_vol = max(
        sum(i["volume_cm3"] for i in items)
        for items in stop_map.values()
    )
    rec = next((b for b in box_types if b["capacity_cm3"] >= max_stop_vol), box_types[-1])

    return {
        "total_items": sum(int(r[1] or 1) for r in rows),
        "total_volume_l": round(total_vol / 1000, 2),
        "missing_dimension_items": missing,
        "all_dims_present": len(missing) == 0,
        "stops": stop_summary,
        "box_plan": rendered_boxes,
        "recommended_box_type": rec["name"],
    }
```

from services.cooler_box_planner import pre_pick_estimate
estimates = {}
for route in routes:
    try:
        estimates[(route["route_id"], str(route["delivery_date"]))] = (
            pre_pick_estimate(route["route_id"], route["delivery_date"])
        )
    except Exception as exc:
        logger.warning("pre_pick_estimate failed for route %s: %s", route["route_id"], exc)
        estimates[(route["route_id"], str(route["delivery_date"]))] = None

return render_template("cooler/route_list.html", routes=routes, estimates=estimates)

## Summary of what these changes do

| Change | Effect |
|---|---|
| `pre_pick_estimate()` | Calculates box requirements from all queue rows before picking starts |
| Route list cards | Manager sees recommended box sizes and missing dimension warnings before opening a route |
| Overfull step badge | "Assign to Boxes" turns red if any box exceeds 100% fill — impossible to miss |
| Overfull banner | Clear instruction to delete and regenerate with a larger box type |
| Print packing list | After picking, one click prints a per-box item list so the packer sorts items without staring at a screen |

No database changes required. No new routes or URL changes.
