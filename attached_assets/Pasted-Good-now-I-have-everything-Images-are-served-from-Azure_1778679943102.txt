cooler_stop_info = {}   # {seq_no (float): {'box_no': int, 'item_count': int}}

    from sqlalchemy import text
    box_rows = db.session.execute(text("""
        SELECT cb.box_no,
               cb.first_stop_sequence,
               cb.last_stop_sequence,
               COUNT(cbi.id) AS item_count
        FROM cooler_boxes cb
        LEFT JOIN cooler_box_items cbi ON cbi.cooler_box_id = cb.id
        WHERE cb.route_id = :rid
          AND cb.status NOT IN ('cancelled')
        GROUP BY cb.id, cb.box_no, cb.first_stop_sequence, cb.last_stop_sequence
        ORDER BY cb.box_no
    """), {"rid": route_id}).fetchall()

    for r in box_rows:
        if r[1] is not None and r[2] is not None:
            first_s = float(r[1])
            last_s  = float(r[2])
            box_no  = r[0]
            item_count = int(r[3] or 0)
            for sd in stops_data:
                seq = float(sd['stop'].seq_no)
                if first_s <= seq <= last_s:
                    # A stop may span multiple boxes — accumulate
                    if seq not in cooler_stop_info:
                        cooler_stop_info[seq] = []
                    cooler_stop_info[seq].append({
                        'box_no': box_no,
                        'item_count': item_count,
                    })

python# Cooler items for this stop (gated by flag)
cooler_items_for_stop = []
if is_driver_view_enabled() and invoice_nos:
    from sqlalchemy import text
    placeholders = ','.join(f"'{n}'" for n in invoice_nos)
    cooler_rows = db.session.execute(text(f"""
        SELECT cbi.invoice_no,
               cbi.item_code,
               COALESCE(cbi.qty_assigned, bpq.qty_required, 0) AS qty,
               cb.box_no,
               COALESCE(psi.item_name, cbi.item_code) AS item_name,
               COALESCE(psi.item_description, '') AS item_description
        FROM cooler_box_items cbi
        JOIN cooler_boxes cb ON cb.id = cbi.cooler_box_id
        LEFT JOIN batch_pick_queue bpq
               ON bpq.invoice_no = cbi.invoice_no
              AND bpq.item_code  = cbi.item_code
              AND bpq.pick_zone_type = 'cooler'
        LEFT JOIN ps_items_dw psi ON psi.item_code_365 = cbi.item_code
        WHERE cbi.invoice_no IN ({placeholders})
          AND cb.route_id = :rid
          AND cb.status NOT IN ('cancelled')
        ORDER BY cb.box_no, cbi.item_code
    """), {"rid": route.id}).fetchall()

    for r in cooler_rows:
        cooler_items_for_stop.append({
            'invoice_no':      r[0],
            'item_code':       r[1],
            'qty':             int(r[2] or 0),
            'box_no':          r[3],
            'item_name':       r[4],
            'item_description': r[5],
            # Image served from static cache — same pattern as picking screen
            'image_path':      f"images/{r[1]}.webp",
            'image_fallback':  "images/image-not-found.png",
        })

Image availability note
Images are fetched from Azure blob storage on demand and cached locally at static/images/{item_code}.webp by image_handler.py. The onerror fallback handles missing images gracefully. No new image fetching logic is needed — the existing cache already works.
If you want to pre-warm the image cache for cooler items when the route starts (so images load instantly in the field), optionally call prefetch_images_for_invoice() from image_handler.py in start_route():
python# In start_route(), after marking route IN_TRANSIT:
from image_handler import prefetch_images_for_invoice
cooler_codes = [r[0] for r in db.session.execute(text("""
    SELECT DISTINCT cbi.item_code
    FROM cooler_box_items cbi
    JOIN cooler_boxes cb ON cb.id = cbi.cooler_box_id
    WHERE cb.route_id = :rid
"""), {"rid": route_id}).fetchall()]
if cooler_codes:
    prefetch_images_for_invoice(cooler_codes)  # runs in background thread
