    except Exception as e:

1.2 Add SQLAlchemy model

File: models.py (near other WMS OI models)

class WmsDynamicRule(db.Model):
    __tablename__ = "wms_dynamic_rules"

    id = db.Column(db.Integer, primary_key=True, autoincrement=True)
    name = db.Column(db.String(120), nullable=False)

    # Must match the resolver attr names (no wms_ prefix)
    target_attr = db.Column(db.String(64), nullable=False)      # e.g. 'pressure_sensitivity'
    action_value = db.Column(db.String(100), nullable=False)    # stored as string; cast on apply
    confidence = db.Column(db.Integer, nullable=False, default=65)
    priority = db.Column(db.Integer, nullable=False, default=100)

    stop_processing = db.Column(db.Boolean, nullable=False, default=True)
    is_active = db.Column(db.Boolean, nullable=False, default=True)

    condition_json = db.Column(db.Text, nullable=False)
    notes = db.Column(db.Text, nullable=True)
    updated_by = db.Column(db.String(100), nullable=True)
    updated_at = db.Column(UTCDateTime(), nullable=True, default=get_utc_now)

    def __repr__(self):
        return f"<WmsDynamicRule {self.id} {self.target_attr} prio={self.priority}>"

3.2 Update resolve_attribute signature (safe approach)

Keep existing positional calls intact by adding dynamic params at the end as keyword-only usage:

def resolve_attribute(..., threshold: int = 60,
                      dynamic_value: Any = None,
                      dynamic_conf: int = 0,
                      dynamic_reason: str = "",
                      dynamic_rule_id: int = None,
                      dynamic_rule_name: str = None) -> Tuple[Any,int,str,str,dict]:


Return five values now: (final_value, confidence, reason, source, meta)
Where meta is a dict like {"rule_id":..., "rule_name":...} for dynamic rules, otherwise {}.

Logic:

if manual override → source MANUAL

elif dynamic_value is not None → source DYNAMIC_RULE (confidence = dynamic_conf, reason includes rule name/id)

elif category default → source CATEGORY_DEFAULT

elif computed_conf >= threshold → source RULES

else → None + ambiguous

3.3 Update helpers

In determine_class_source(evidence) update overall source precedence to:

MANUAL > DYNAMIC_RULE > CATEGORY_DEFAULT > RULES

In build_class_notes(...), add “Uses dynamic rules” when present.
