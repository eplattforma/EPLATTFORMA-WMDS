function toggleHdrDetails() {
  const el = document.getElementById('hdr-details');
  const caret = document.getElementById('hdr-details-caret');
  if (!el) return;
  const show = (el.style.display === 'none' || el.style.display === '');
  el.style.display = show ? 'block' : 'none';
  if (caret) caret.textContent = show ? '▴' : '▾';
}

3) Update JS so the header shows “COLLECT” and the card shows “REMAINING”
3.1 Update your refresh function

Where you currently set:

hdr-inv, hdr-ex, hdr-entered, hdr-collect

and you also compute due / variance

Change the logic to:

collectTarget = max(0, invoiceTotal - exceptionsTotal)

entered = collected (0 if online/credit)

remaining = max(0, collectTarget - entered)

Add these assignments:

function refreshAmountsUI() {
  const inv = totalInvoicesAmount;
  const exTot = exceptions.reduce((sum, ex) => sum + (ex.exception_value || 0), 0);
  const collectTarget = Math.max(0, inv - exTot);

  const method = document.getElementById('cod-method')?.value || '';
  const entered = (!isCredit && method !== 'online') ? (parseFloat(document.getElementById('cod-received')?.value) || 0) : 0;
  const remaining = Math.max(0, collectTarget - entered);

  // HEADER (target only)
  document.getElementById('hdr-collect').textContent = collectTarget.toFixed(2);

  // HEADER DETAILS (hidden by default)
  const hdrInv = document.getElementById('hdr-inv'); if (hdrInv) hdrInv.textContent = inv.toFixed(2);
  const hdrEx = document.getElementById('hdr-ex'); if (hdrEx) hdrEx.textContent = exTot.toFixed(2);
  const hdrEnt = document.getElementById('hdr-entered'); if (hdrEnt) hdrEnt.textContent = entered.toFixed(2);

  // PAYMENT CARD MINI (if you add it)
  const invMini = document.getElementById('cod-inv-mini'); if (invMini) invMini.textContent = inv.toFixed(2);
  const exMini = document.getElementById('cod-ex-mini'); if (exMini) exMini.textContent = exTot.toFixed(2);

  // PAYMENT CARD “Remaining”
  const remEl = document.getElementById('cod-remaining');
  if (remEl) remEl.textContent = `€ ${remaining.toFixed(2)}`;
}

Then call refreshAmountsUI() at the end of:

your exception add/remove/render function

keypad input handler (kp, kpBack, kpClear, kpExact)

payment method change handler

cheque date change handler

This ensures the target and remaining are always right.
