1.3 Add JS: compute when signature is required + update UI

Add this in the same JS block where you handle payment method and validation:

function _getPaymentMethod() {
  // Your UI uses tiles but you also have a hidden <select id="cod-method"> in most builds.
  const sel = document.getElementById('cod-method');
  return (sel ? sel.value : '').toLowerCase();
}

function _getChequeDateStr() {
  // Use your existing cheque date input id. If yours is different, change it here.
  const el = document.getElementById('cheque-date');
  return el ? el.value : '';
}

function isSignatureRequired() {
  if (window.isCreditCustomer === true) return true;

  const method = _getPaymentMethod();

  if (method === 'online') return true;

  if (method === 'cheque') {
    const ds = _getChequeDateStr();
    if (!ds) return false;
    const today = new Date(); today.setHours(0,0,0,0);
    const d = new Date(ds + 'T00:00:00');
    if (d > today) return true; // post-dated cheque
  }

  return false;
}

function updateSignatureRequirementUI() {
  const required = isSignatureRequired();
  const badge = document.getElementById('sig-required-badge');
  const help = document.getElementById('sig-help');
  const warn = document.getElementById('sig-warn');
  const sw = document.getElementById('signedInvoiceSwitch');
  const block = document.getElementById('sigBlock');

  if (!badge || !help || !warn || !sw || !block) return;

  if (required) {
    badge.textContent = 'REQUIRED';
    badge.className = 'badge bg-danger ms-2';
    help.textContent = 'Required for Credit / Pay Online / Post-dated cheque.';
    warn.style.display = sw.checked ? 'none' : 'block';
    block.classList.toggle('border-danger', !sw.checked);
  } else {
    badge.textContent = 'Optional';
    badge.className = 'badge bg-secondary ms-2';
    help.textContent = 'Recommended, but optional for Cash/Day Cheque/Card.';
    warn.style.display = 'none';
    block.classList.remove('border-danger');
  }

  // Hook into your existing “Complete Delivery” enable/disable function if you have one
  if (typeof updateCompleteDeliveryButton === 'function') {
    updateCompleteDeliveryButton();
  }
}

function validateBeforeSubmit() {
  const required = isSignatureRequired();
  const sw = document.getElementById('signedInvoiceSwitch');
  if (required && sw && !sw.checked) {
