2) Add helper methods in run_service.py
========================================

In services/forecast/run_service.py add helpers so run status can be updated safely and consistently.

Add something like:

from datetime import datetime
from sqlalchemy import text

def _utcnow():
    return datetime.utcnow()

def _heartbeat(run_id, step=None, note=None):
    payload = {
        "id": run_id,
        "hb": _utcnow(),
        "step": step,
        "note": note,
    }
    db.session.execute(text("""
        UPDATE forecast_runs
        SET last_heartbeat_at = :hb,
            current_step = COALESCE(:step, current_step),
            progress_note = COALESCE(:note, progress_note)
        WHERE id = :id
    """), payload)
    db.session.commit()

def _mark_run_finished(run_id, status, note=None, completed_at=None):
    if completed_at is None:
        completed_at = _utcnow()
    db.session.execute(text("""
        UPDATE forecast_runs
        SET status = :status,
            notes = COALESCE(:note, notes),
            completed_at = :completed_at,
            last_heartbeat_at = :completed_at
        WHERE id = :id
    """), {
        "id": run_id,
        "status": status,
        "note": note,
        "completed_at": completed_at,
    })
    db.session.commit()

Do NOT rely only on in-memory ORM state for long-running run tracking. Use committed updates for heartbeat/status.
