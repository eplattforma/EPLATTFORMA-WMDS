# ai_report_feedback.py
import os, json, hashlib
from flask import Blueprint, request, jsonify
from flask_login import login_required, current_user
from openai import OpenAI

bp = Blueprint("ai_report_feedback", __name__, url_prefix="/api/ai")

SYSTEM_PROMPT = """
You are a B2B sales analytics assistant for STEP EPLATTFORMA.
You will receive a JSON payload describing a report (KPIs, period, peer group, trends, lists).
Rules:
- Use ONLY the provided numbers. If something is missing, say "insufficient data".
- Be practical for a sales rep: what to say, what to do next.
- Keep English terminology used in the report (e.g., Net Sales, Peer Avg, Price Index).
- Output MUST follow the provided JSON schema strictly.
"""

# JSON Schema the model must return (stable output for your UI)
INSIGHT_SCHEMA = {
  "name": "report_feedback",
  "schema": {
    "type": "object",
    "additionalProperties": False,
    "properties": {
      "executive_summary": {"type": "string"},
      "key_findings": {"type": "array", "items": {"type": "string"}},
      "opportunities": {"type": "array", "items": {"type": "string"}},
      "risks": {"type": "array", "items": {"type": "string"}},
      "questions_to_validate": {"type": "array", "items": {"type": "string"}},
      "recommended_actions": {"type": "array", "items": {"type": "string"}},
      "chart_notes": {"type": "array", "items": {"type": "string"}}
    },
    "required": [
      "executive_summary","key_findings","opportunities",
      "risks","questions_to_validate","recommended_actions","chart_notes"
    ]
  },
  "strict": True
}

def _is_allowed():
    return getattr(current_user, "role", None) in ["admin", "sales_manager", "warehouse_manager"]

@bp.post("/report-feedback")
@login_required
def report_feedback():
    if not _is_allowed():
        return jsonify({"error": "forbidden"}), 403

    payload = request.get_json(silent=True) or {}
    # Basic guardrails
    raw = json.dumps(payload, ensure_ascii=False)
    if len(raw) > 250_000:
        return jsonify({"error": "payload too large; send aggregates/top-N only"}), 400

    client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
    model = os.getenv("OPENAI_MODEL", "gpt-5.2")

    resp = client.responses.create(
        model=model,
        input=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": raw}
        ],
        # keep it concise; adjust as you like
        text={
            "verbosity": "low",
            "format": {"type": "json_schema", "json_schema": INSIGHT_SCHEMA}
        },
        # prevents storing the interaction as a “stored conversation”
        store=False,
        metadata={"app": "wmds", "feature": "report_feedback"}
    )

    # The SDK returns text; parse JSON
    out_text = resp.output_text or "{}"
    try:
        out = json.loads(out_text)
    except Exception:
        out = {"executive_summary": out_text, "key_findings": [], "opportunities": [],
               "risks": [], "questions_to_validate": [], "recommended_actions": [], "chart_notes": []}

    return jsonify(out)
This uses the Responses API and Structured Outputs (JSON schema) so your UI is deterministic.

import os, json, hashlib
from datetime import datetime, timedelta, timezone

from openai import OpenAI
from sqlalchemy import text
from app import db

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
OPENAI_MODEL = os.getenv("OPENAI_MODEL", "").strip()

CACHE_TTL_HOURS = 12
MAX_ROWS = 80  # keep payload compact

JSON_SCHEMA = {
  "type": "object",
  "additionalProperties": False,
  "properties": {
    "summary": {"type": "string"},
    "peer_context": {"type": "string"},
    "key_findings": {"type": "array", "items": {"type": "string"}},
    "risks": {"type": "array", "items": {"type": "string"}},
    "opportunities": {
      "type": "array",
      "items": {
        "type": "object",
        "additionalProperties": False,
        "properties": {
          "title": {"type": "string"},
          "why": {"type": "string"},
          "expected_impact": {"type": "string"},
          "confidence": {"type": "number"}
        },
        "required": ["title", "why", "expected_impact", "confidence"]
      }
    },
    "next_actions": {
      "type": "array",
      "items": {
        "type": "object",
        "additionalProperties": False,
        "properties": {
          "priority": {"type": "string", "enum": ["P0", "P1", "P2"]},
          "action": {"type": "string"},
          "script_hint": {"type": "string"}
        },
        "required": ["priority", "action", "script_hint"]
      }
    }
  },
  "required": ["summary", "peer_context", "key_findings", "risks", "opportunities", "next_actions"]
}


def _hash_payload(payload: dict) -> str:
    s = json.dumps(payload, sort_keys=True, ensure_ascii=False)
    return hashlib.sha256(s.encode("utf-8")).hexdigest()


def _clip_payload(payload: dict) -> dict:
    """Keep arrays small to avoid huge token usage."""
    p = dict(payload)
    for k in ["white_space", "lapsed_items", "top_items", "price_outliers"]:
        if isinstance(p.get(k), list):
            p[k] = p[k][:MAX_ROWS]
    return p


def get_cached_feedback(payload_hash: str):
    q = text("""
      SELECT response_json
      FROM ai_feedback_cache
      WHERE payload_hash = :h AND expires_at > NOW()
      LIMIT 1
    """)
    row = db.session.execute(q, {"h": payload_hash}).fetchone()
    return row[0] if row else None


def save_cached_feedback(payload_hash: str, response_json: dict):
    q = text("""
      INSERT INTO ai_feedback_cache(payload_hash, expires_at, response_json)
      VALUES (:h, :exp, :j)
      ON CONFLICT (payload_hash) DO UPDATE
      SET expires_at = EXCLUDED.expires_at,
          response_json = EXCLUDED.response_json
    """)
    exp = datetime.now(timezone.utc) + timedelta(hours=CACHE_TTL_HOURS)
    db.session.execute(q, {"h": payload_hash, "exp": exp, "j": json.dumps(response_json)})
    db.session.commit()


def generate_feedback(report_snapshot: dict, language: str = "el") -> dict:
    if not OPENAI_MODEL:
        raise ValueError("OPENAI_MODEL is not set")

    snapshot = _clip_payload(report_snapshot)
    payload_hash = _hash_payload(snapshot)

    cached = get_cached_feedback(payload_hash)
    if cached:
        # cached is JSONB (already dict-like in many drivers). If string, json.loads it.
        return cached if isinstance(cached, dict) else json.loads(cached)

    # Prompt: Greek narrative, keep metric names in English as in the report
    system = (
      "You are a sales analytics assistant for a B2B wholesaler. "
      "You receive a compact JSON snapshot of a Customer vs Peer Group dashboard. "
      "Return actionable insights for a sales rep. "
      "Write in Greek, but keep the metric/section names in English exactly as provided (e.g., Sales Index, Price Index). "
      "Be concrete and reference numbers from the snapshot."
    )

    user = json.dumps(snapshot, ensure_ascii=False)

    # Structured Outputs via JSON Schema (strict)
    resp = client.responses.create(
        model=OPENAI_MODEL,
        input=[
          {"role": "system", "content": [{"type": "input_text", "text": system}]},
          {"role": "user", "content": [{"type": "input_text", "text": user}]},
        ],
        text={
          "format": {
            "type": "json_schema",
            "name": "customer_peer_feedback",
            "schema": JSON_SCHEMA,
            "strict": True
          }
        },
        store=False
    )

    out = json.loads(resp.output_text)
    save_cached_feedback(payload_hash, out)
    return out
Notes:
•	client.responses.create(...) usage is per OpenAI docs. 
•	text.format with json_schema + strict is the Structured Outputs pattern. 
•	store: false is supported on Responses create.

def feedback():

Two quick clarifications (from your earlier questions)
What is “Min Peer Penetration”?
It’s the minimum % of peers that must buy an item for it to qualify as a “peer staple” in White Space.
•	Example: peers_count = 50, min_peer_penetration = 30%
→ item must be bought by ≥ 15 peers to appear as “Peers buy it, customer doesn’t”.
Your “Monthly Trend: Customer vs Peer Avg” graph
•	Blue line = this customer’s Net Sales per month (often net of returns)
•	Purple dotted = Peer Avg Sales / Customer per month
If blue is consistently below purple, the customer is underperforming vs similar accounts; if it spikes then drops, it’s “bulk purchase then inactive” behavior (a rep follow-up trigger).

Power BI measures (για τα tables που έχεις: DimCustomers, DimDates, DimProducts, DimStores, FactSales)
Αφού είπες “looks like returns are subtracted”, ξεκινάμε με assumption ότι τα Sales Return είναι αρνητικά στο line_total_excl. Αν δεν είναι, σου δίνω και εναλλακτικό.
-- Core
Net Sales :=
SUM ( FactSales[line_total_excl] )

Gross Sales (no returns) :=
CALCULATE ( [Net Sales], FactSales[invoice_type] <> "SALES RETURN" )

Returns (value) :=
CALCULATE ( [Net Sales], FactSales[invoice_type] = "SALES RETURN" )   -- συνήθως αρνητικό

Returns (abs) :=
ABS ( [Returns (value)] )

Orders :=
DISTINCTCOUNT ( FactSales[invoice_no] )

Total Qty :=
SUM ( FactSales[quantity] )

Unique Items :=
DISTINCTCOUNT ( FactSales[item_code] )

AOV :=
DIVIDE ( [Net Sales], [Orders] )

Unit Price (avg) :=
DIVIDE ( [Net Sales], [Total Qty] )

Active Customers :=
DISTINCTCOUNT ( FactSales[customer_code] )

-- Time intel (με DimDates[date_key] ως Date table)
Net Sales PY :=
CALCULATE ( [Net Sales], SAMEPERIODLASTYEAR ( DimDates[date_key] ) )

YoY % :=
DIVIDE ( [Net Sales] - [Net Sales PY], [Net Sales PY] )

Net Sales Prev Month :=
CALCULATE ( [Net Sales], DATEADD ( DimDates[date_key], -1, MONTH ) )

MoM % :=
DIVIDE ( [Net Sales] - [Net Sales Prev Month], [Net Sales Prev Month] )
Αν τα returns δεν είναι αρνητικά και έρχονται ως θετικά lines με invoice_type="SALES RETURN", τότε:
Net Sales (forced) :=
VAR sales =
    CALCULATE ( SUM ( FactSales[line_total_excl] ), FactSales[invoice_type] <> "SALES RETURN" )
VAR returns =
    CALCULATE ( SUM ( FactSales[line_total_excl] ), FactSales[invoice_type] = "SALES RETURN" )
RETURN
sales - returns
Peer measures (αν έχεις DimCustomers[reporting_group] ή peer group field):
Peer Count :=
VAR grp = SELECTEDVALUE ( DimCustomers[reporting_group] )
VAR cust = SELECTEDVALUE ( DimCustomers[customer_code] )
RETURN
CALCULATE (
    DISTINCTCOUNT ( DimCustomers[customer_code] ),
    FILTER ( ALL ( DimCustomers ),
        DimCustomers[reporting_group] = grp &&
        DimCustomers[customer_code] <> cust
    )
)

Peer Net Sales :=
VAR grp = SELECTEDVALUE ( DimCustomers[reporting_group] )
VAR cust = SELECTEDVALUE ( DimCustomers[customer_code] )
RETURN
CALCULATE (
    [Net Sales],
    FILTER ( ALL ( DimCustomers ),
        DimCustomers[reporting_group] = grp &&
        DimCustomers[customer_code] <> cust
    )
)

Peer Avg Sales/Cust :=
DIVIDE ( [Peer Net Sales], [Peer Count] )

Sales Index (vs peer avg = 1.0) :=
DIVIDE ( [Net Sales], [Peer Avg Sales/Cust] )

Peer Unit Price :=
VAR grp = SELECTEDVALUE ( DimCustomers[reporting_group] )
VAR cust = SELECTEDVALUE ( DimCustomers[customer_code] )
VAR peer_sales =
    CALCULATE (
        SUM ( FactSales[line_total_excl] ),
        FILTER ( ALL ( DimCustomers ),
            DimCustomers[reporting_group] = grp &&
            DimCustomers[customer_code] <> cust
        )
    )
VAR peer_qty =
    CALCULATE (
        SUM ( FactSales[quantity] ),
        FILTER ( ALL ( DimCustomers ),
            DimCustomers[reporting_group] = grp &&
            DimCustomers[customer_code] <> cust
        )
    )
RETURN
DIVIDE ( peer_sales, peer_qty )

Price Index (gross) :=
DIVIDE ( [Unit Price (avg)], [Peer Unit Price] )

import os

from flask import Blueprint, request, jsonify
from flask_login import login_required, current_user
from app import db

from services_ai_insights import build_snapshot_for_tab, generate_ai_insight, make_cache_key

bp_ai = Blueprint("ai_insights", __name__, url_prefix="/admin/customer-benchmark/api")

TTL_MIN = int(os.getenv("AI_INSIGHTS_TTL_MIN", "1440"))

def _allowed():
    return getattr(current_user, "role", None) in ("admin", "sales_manager", "sales")

@bp_ai.get("/ai-insights")
@login_required
def get_ai_insights():
    if not _allowed():
        return jsonify({"error": "Access denied"}), 403

    customer_code = request.args.get("customer_code", "").strip()
    peer_key = request.args.get("peer_key", "").strip()
    tab = request.args.get("tab", "").strip()
    date_from = request.args.get("date_from")
    date_to = request.args.get("date_to")
    compare_mode = request.args.get("compare_mode")

    cache_key = make_cache_key(customer_code, peer_key, tab, date_from, date_to, compare_mode)
    row = AiInsightCache.query.filter_by(cache_key=cache_key).first()

    if not row:
        return jsonify({"found": False})

    if row.created_at < (datetime.utcnow() - timedelta(minutes=TTL_MIN)):
        return jsonify({"found": False, "stale": True})

    return jsonify({"found": True, "insight": row.insight_json, "created_at": row.created_at.isoformat()})

@bp_ai.post("/ai-insights")
@login_required
def refresh_ai_insights():
    if not _allowed():
        return jsonify({"error": "Access denied"}), 403

    payload = request.get_json(force=True) or {}
    customer_code = (payload.get("customer_code") or "").strip()
    peer_key = (payload.get("peer_key") or "").strip()
    tab = (payload.get("tab") or "").strip()
    date_from = payload.get("date_from")
    date_to = payload.get("date_to")
    compare_mode = payload.get("compare_mode")

    # Build snapshot on server (do NOT trust client rows)
    snapshot = build_snapshot_for_tab(
        customer_code=customer_code,
        peer_key=peer_key,
        tab=tab,
        date_from=date_from,
        date_to=date_to,
        compare_mode=compare_mode,
        params=payload.get("params") or {}
    )

    insight = generate_ai_insight(tab=tab, snapshot=snapshot)

    cache_key = make_cache_key(customer_code, peer_key, tab, date_from, date_to, compare_mode)
    row = AiInsightCache.query.filter_by(cache_key=cache_key).first()
    if not row:
        row = AiInsightCache(
            cache_key=cache_key,
            customer_code=customer_code,
            peer_key=peer_key,
            tab=tab,
            date_from=snapshot["context"]["date_from"],
            date_to=snapshot["context"]["date_to"],
            compare_mode=compare_mode,
        )
        db.session.add(row)

    row.snapshot_json = snapshot
    row.insight_json = insight
    row.model = os.getenv("OPENAI_MODEL", "")
    row.created_at = datetime.utcnow()

    db.session.commit()
    return jsonify({"ok": True, "insight": insight})

import os, json, hashlib
from datetime import date
from openai import OpenAI

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY", ""))

PROMPT_VERSION = "v1"

BASE_SCHEMA = {
  "type": "object",
  "additionalProperties": False,
  "properties": {
    "headline": {"type": "string"},
    "insights": {
      "type": "array",
      "items": {
        "type": "object",
        "additionalProperties": False,
        "properties": {
          "title": {"type": "string"},
          "detail": {"type": "string"},
          "why_it_matters": {"type": "string"},
          "action": {"type": "string"},
          "confidence": {"type": "number"}
        },
        "required": ["title","detail","why_it_matters","action","confidence"]
      }
    },
    "talk_track": {"type": "array", "items": {"type": "string"}},
    "data_checks": {"type": "array", "items": {"type": "string"}},
    "priority_items": {
      "type": "array",
      "items": {
        "type": "object",
        "additionalProperties": False,
        "properties": {
          "item_code": {"type": "string"},
          "item_name": {"type": "string"},
          "metric": {"type": "string"},
          "value": {"type": "number"},
          "next_step": {"type": "string"}
        },
        "required": ["item_code","item_name","metric","value","next_step"]
      }
    }
  },
  "required": ["headline","insights","talk_track","data_checks","priority_items"]
}

def make_cache_key(customer_code, peer_key, tab, date_from, date_to, compare_mode):
    raw = f"{customer_code}|{peer_key}|{tab}|{date_from}|{date_to}|{compare_mode}|{PROMPT_VERSION}"
    return hashlib.sha1(raw.encode("utf-8")).hexdigest()

def _tab_instructions(tab: str) -> str:
    return {
      "white_space": "Focus on items peers buy but customer doesn't. Explain min_peer_penetration.",
      "lapsed": "Focus on items customer bought before but not recently; suggest reactivation.",
      "category_mix": "Focus on under-index categories vs peers; cross-sell.",
      "price_vs_peers": "Focus on overpay/underpay outliers; churn risk vs margin leakage.",
      "item_recency": "Focus on recency/frequency signals per item; reorder cadence."
    }.get(tab, "Focus on actionable benchmarking insights.")

def generate_ai_insight(tab: str, snapshot: dict) -> dict:
    model = os.getenv("OPENAI_MODEL", "")
    if not model:
        raise RuntimeError("OPENAI_MODEL missing")

    system = (
      "You are a B2B sales analytics assistant. "
      "Return JSON ONLY. Use English field names. "
      "Write the narrative text (headline/detail/action/talk_track) in Greek, "
      "but keep business terms in English (AOV, Price Index, peer penetration, etc)."
    )

    user = {
      "tab": tab,
      "instructions": _tab_instructions(tab),
      "snapshot": snapshot
    }

    resp = client.responses.create(
      model=model,
      input=[
        {"role": "system", "content": system},
        {"role": "user", "content": json.dumps(user, ensure_ascii=False)}
      ],
      text={
        "format": {
          "type": "json_schema",
          "name": "customer_peer_tab_insight",
          "schema": BASE_SCHEMA,
          "strict": True
        }
      },
      # If you don't want OpenAI to store outputs, explicitly disable:
      store=False
    )
    # With structured outputs, the response text is valid JSON matching schema
    return json.loads(resp.output_text)
Notes:
•	text.format + strict: true είναι το key για “hard schema”. 
•	Το store μπορεί να μπει false για να μη γίνεται storage από πλευράς API.

async function aiLoad(tab) {
  const url = `/admin/customer-benchmark/api/ai-insights?customer_code=${encodeURIComponent(selectedCustomer)}&peer_key=${encodeURIComponent(peerKey)}&tab=${encodeURIComponent(tab)}&date_from=${dateFrom}&date_to=${dateTo}&compare_mode=${compareMode}`;
  const r = await fetch(url);
  const j = await r.json();
  if (!j.found) {
    document.getElementById('ai_body').innerHTML = '<div class="muted">No AI insights yet.</div>';
    return;
  }
  renderAi(j.insight, j.created_at);
}

async function aiRefresh(tab) {
  document.getElementById('ai_body').innerHTML = '<div class="muted">Generating…</div>';
  const r = await fetch('/admin/customer-benchmark/api/ai-insights', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({
      customer_code: selectedCustomer,
      peer_key: peerKey,
      tab,
      date_from: dateFrom,
      date_to: dateTo,
      compare_mode: compareMode,
      params: currentTabParams()   // e.g. min_peer_penetration etc
    })
  });
  const j = await r.json();
  if (!r.ok) return alert(j.error || 'AI error');
  renderAi(j.insight);
}

function renderAi(insight, createdAt=null) {
  const meta = createdAt ? `Last updated: ${createdAt}` : '';
  document.getElementById('ai_meta').textContent = meta;

  const blocks = [];
  blocks.push(`<div class="ai-headline">${escapeHtml(insight.headline)}</div>`);

  blocks.push('<ul class="ai-list">');
  (insight.insights || []).forEach(x => {
    blocks.push(`<li><b>${escapeHtml(x.title)}</b><br>${escapeHtml(x.detail)}<br><i>${escapeHtml(x.action)}</i></li>`);
  });
  blocks.push('</ul>');

  blocks.push('<div class="ai-sub">Talk track</div><ul>');
  (insight.talk_track || []).forEach(t => blocks.push(`<li>${escapeHtml(t)}</li>`));
  blocks.push('</ul>');

  document.getElementById('ai_body').innerHTML = blocks.join('');
}
