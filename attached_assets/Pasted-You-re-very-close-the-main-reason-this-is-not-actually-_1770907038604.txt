Task 2 — Make the date-range code robust in peer blueprint

Your peer_bp._resolve_range() currently doesn’t swap dates and can return invalid values.

Replace it with the same safe version you used in catmgr:

def _resolve_range(args):
    preset = (args.get("preset") or "last90").lower().strip()
    today = date.today()

    if preset == "last30":
        return today - timedelta(days=29), today
    if preset in ("last90", ""):
        return today - timedelta(days=89), today

    d_from = _parse_date(args.get("from"))
    d_to = _parse_date(args.get("to"))
    if not d_from or not d_to:
        return today - timedelta(days=89), today
    if d_to < d_from:
        d_from, d_to = d_to, d_from
    return d_from, d_to

Task 3 — Fix PY compare shifting (peer blueprint)

Your _compare_range() “except ValueError” branch is wrong (it forces month=2/day=28 always).

Replace with safe clamp logic:

def _safe_shift_year(d, years):
    try:
        return d.replace(year=d.year + years)
    except ValueError:
        # handles Feb 29
        return d.replace(year=d.year + years, month=2, day=28)

def _compare_range(d_from, d_to, mode):
    mode = (mode or "").lower().strip()
    if mode not in ("prev", "py"):
        return None
    if mode == "py":
        return _safe_shift_year(d_from, -1), _safe_shift_year(d_to, -1)

    length = (d_to - d_from).days + 1
    prev_end = d_from - timedelta(days=1)
    prev_start = prev_end - timedelta(days=length - 1)
    return prev_start, prev_end
